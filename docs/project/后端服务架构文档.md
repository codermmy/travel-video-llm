```python
# 依赖注入的 get_db() 生成器
def get_db() -> Generator[Session, None, None]:
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

### 2. 缓存服务

**技术栈**: 内存缓存（SimpleCache）+ Redis（Celery 后端）

#### 内存缓存

**文件**: `backend/app/utils/cache.py`

```python
# 线程安全的通用缓存类
class SimpleCache(Generic[T]):
    def __init__(self) -> None:
        self._lock = Lock()
        self._store: dict[str, CacheEntry[T]] = {}

    def get(self, key: str) -> Optional[T]:
        # 自动过期检查
        entry = self._store.get(key)
        if entry.expires_at <= now:
            self._store.pop(key, None)
            return None
        return entry.value

    def set(self, key: str, value: T, ttl: int = 3600) -> None:
        expires_at = time.time() + ttl
        self._store[key] = CacheEntry(value=value, expires_at=expires_at)
```

**专用缓存实例**:
```python
# 地理编码缓存（TTL: 24小时）
geocoding_cache: SimpleCache[str] = SimpleCache()

# AI 结果缓存（TTL: 24小时）
_ai_cache: SimpleCache[object] = SimpleCache()

# 统一缓存函数（AI 服务使用）
def simple_cache(key: str, value: Optional[object] = None, ttl: int = 86400) -> object | None:
    if value is None:
        return _ai_cache.get(key)
    _ai_cache.set(key, value, ttl=ttl)
    return value
```

**缓存策略**:
| 缓存类型 | TTL | 缓存键格式 | 使用场景 |
|---------|-----|-----------|---------|

| AI 故事 | 86400秒（24小时） | `ai:{provider}:story:{event_id}` | 事件故事结果 |

**任务流程示例**（`process_new_photos_task`）:

```python
@celery_app.task(name="process_new_photos_task")
def process_new_photos_task(user_id: str, photo_ids: list[str], task_record_id: str) -> dict:
    db: Session = SessionLocal()
    try:
        task = _get_task(db, task_record_id, user_id)
        if not task:
            raise RuntimeError("task record not found")

        # 阶段1: 聚类（10%进度）
        _update_task(db, task, status="started", stage="clustering", progress=10)
        created = cluster_user_photos(user_id=user_id, db=db)
        _update_task(db, task, progress=70)

        # 阶段2: 地理编码（90%进度）
        geocoding_service.update_event_locations(user_id=user_id, db=db)
        _update_task(db, task, stage="geocoding", progress=90)

        # 阶段3: AI 生成（85-99%进度）
        ai_success_count = 0
        ai_failed_count = 0
        if created:
            _update_task(db, task, stage="ai", progress=85)
            for idx, event in enumerate(created, start=1):
                ok, _ = generate_event_story_for_event(db=db, user_id=user_id, event_id=event.id)
                if ok:
                    ai_success_count += 1
                else:
                    ai_failed_count += 1
                progress = 85 + int((idx / total_events) * 14)
                _update_task(db, task, stage="ai", progress=min(progress, 99))

        # 完成（100%进度）
        result = {
            "photos": len(photo_ids),
            "events": len(created),
            "updatedLocations": updated_locations,
            "aiGenerated": ai_success_count,
            "aiFailed": ai_failed_count,
        }
        _update_task(db, task, status="success", progress=100, result=result)
        return result
    finally:
        db.close()
```

**任务状态跟踪**（`app/models/task.py`）:

| 状态 | 含义 | 使用场景 |
|------|------|---------|
| `pending` | 任务已创建，等待执行 | Celery 任务入队 |
| `started` | 任务正在执行 | Celery worker 接收 |
| `failure` | 任务执行失败 | 异常或超时 |
| `success` | 任务执行成功 | 所有阶段完成 |

**阶段（stage）**:
| 阶段 | 说明 |
|------|------|
| `clustering` | 照片聚类中 |
| `geocoding` | 地理编码中 |
| `ai` | AI 故事生成中 |

**进度追踪**:
- `progress`: 0-100 整数
- `total`: 预期总工作量（如照片数量）
- `result`: 任务完成后的结构化数据（JSON 字符串）

```python
class SpacetimeClustering:
    def _photo_distance(self, a: PhotoData, b: PhotoData) -> float:
        """计算两张照片的时空距离

        公式：sqrt(time_norm² + spatial_norm²)

        - time_norm: 时间差 / time_threshold（归一化到 [0, 1]）
        - spatial_norm: Haversine 距离 / distance_threshold（归一化到 [0, 1]）
        """
        time_threshold = timedelta(hours=self.config.time_threshold_hours)
        time_diff = abs(a.shoot_time - b.shoot_time)
        time_norm = min(time_diff / time_threshold, 1.0)

        spatial_norm = 0.0
        if (a.gps_lat and a.gps_lon and b.gps_lat and b.gps_lon):
            dist_km = haversine_distance(a.gps_lat, a.gps_lon, b.gps_lat, b.gps_lon)
            spatial_norm = min(dist_km / self.config.distance_threshold_km, 1.0)

        return math.sqrt(time_norm * time_norm + spatial_norm * spatial_norm)
```

```python
def create_events_from_clusters(
    self,
    user_id: str,
    clusters: list[list[PhotoData]],
    db: Session,
) -> list[Event]:
    """从聚类结果创建 Event 记录"""
    for cluster_photos in clusters:
        if len(cluster_photos) < self.config.min_photos_per_event:
            continue  # 跳过小簇

        # 计算时间范围
        sorted_photos = sorted(cluster_photos, key=lambda p: p.shoot_time)
        start_time = sorted_photos[0].shoot_time
        end_time = sorted_photos[-1].shoot_time

        # 计算地理中心点
        gps_points = [(p.gps_lat, p.gps_lon) for p in sorted_photos if p.gps_lat and p.gps_lon]
        center = calculate_center_point(gps_points) if gps_points else None

        # 选择封面照片（中间位置）
        cover_photo = sorted_photos[len(sorted_photos) // 2]

        event = Event(
            user_id=user_id,
            title="",  # 待AI生成
            start_time=start_time,
            end_time=end_time,
            photo_count=len(sorted_photos),
            cover_photo_id=cover_photo.id,
            cover_photo_url=cover_photo.thumbnail_url,
            status="clustered",
        )
        if center:
            event.gps_lat = Decimal(str(center[0]))
            event.gps_lon = Decimal(str(center[1]))

        db.add(event)
        db.flush()  # 获取 event.id

        # 更新照片关联
        for photo in db.scalars(select(Photo).where(Photo.id.in_([p.id for p in sorted_photos]))):
            photo.event_id = event.id
            photo.status = "clustered"

        db.commit()
```

### 5. 地理编码服务

**文件**: `backend/app/services/geocoding_service.py`

**功能**: GPS 坐标 → 地址转换（使用高德地图API）

```python
class GeocodingService:
    def __init__(self) -> None:
        self._ttl_seconds = 86400  # 24小时

    def _cache_key(self, lat: float, lon: float) -> str:
        """生成缓存键：geocode:30.2590,104.1520"""
        return f"geocode:{format_coordinates(lat, lon, decimals=4)}"

    def get_location_name(
        self,
        lat: float,
        lon: float,
    ) -> Optional[str]:
        """获取位置名称（带缓存）"""
        key = self._cache_key(lat, lon)
        cached = geocoding_cache.get(key)
        if cached is not None:
            return cached

        # 调用高德地图 API
        location = amap_client.reverse_geocode(lat=lat, lon=lon)
        if location:
            geocoding_cache.set(key, location, ttl=self._ttl_seconds)
        return location

    def update_event_locations(
        self,
        user_id: str,
        db: Session,
    ) -> int:
        """批量更新事件的 location_name（异步任务调用）"""
        events = db.scalars(
            select(Event).where(
                and_(Event.user_id == user_id, Event.location_name.is_(None))
            )
        ).all()

        updated = 0
        for e in events:
            if e.gps_lat is None or e.gps_lon is None:
                continue

            try:
                name = self.get_location_name(float(e.gps_lat), float(e.gps_lon))
                if name:
                    e.location_name = name
                    updated += 1
            except Exception:
                continue

        if updated:
            db.commit()
        return updated
```

```python
@dataclass(frozen=True)
class StorageUploadResult:
    local_path: str  # 本地路径
    public_url: str  # 公共访问URL
    storage_provider: str  # 'local' 或 'oss'
    object_key: Optional[str]  # OSS 对象键（仅 OSS）

class LocalStorageProvider:
    def __init__(self) -> None:
        self.upload_dir = Path(settings.upload_dir) / 'photos'

    def save_file(self, file_hash: str, file_obj: BinaryIO) -> Path:
        """保存到本地：uploads/photos/{hash}.jpg"""

    def build_public_url(self, file_hash: str) -> str:
        """构建本地 URL：/uploads/photos/{hash}.jpg"""
        return f'{settings.backend_public_base_url_normalized}/uploads/photos/{file_hash}.jpg'

class OssStorageProvider:
    def __init__(self) -> None:
        self.endpoint = settings.oss_endpoint
        self.bucket_name = settings.oss_bucket

    def build_object_key(self, file_hash: str) -> str:
        """构建 OSS 对象键：photos/{hash}.jpg"""

    def upload_file(self, local_path: str, object_key: str) -> None:
        """上传到 OSS（使用 oss2 库）"""

    def build_public_url(self, object_key: str) -> str:
        """构建 OSS 公共 URL：https://{bucket}.{endpoint}/photos/{hash}.jpg"""

class StorageService:
    def __init__(self) -> None:
        self.local = LocalStorageProvider()
        self.oss = OssStorageProvider() if settings.oss_enabled else None

    def upload_photo_file(
        self,
        file_hash: str,
        file_obj: BinaryIO,
    ) -> StorageUploadResult:
        """上传照片文件（本地或 OSS）"""
        # 1. 保存到本地
        local_path = self.local.save_file(file_hash=file_hash, file_obj=file_obj)

        # 2. 如果未启用 OSS，返回本地
        if self.oss is None:
            public_url = self.local.build_public_url(file_hash)
            return StorageUploadResult(
                local_path=str(local_path),
                public_url=public_url,
                storage_provider='local',
                object_key=None,
            )

        # 3. 上传到 OSS
        object_key = self.oss.build_object_key(file_hash)
        self.oss.upload_file(local_path=str(local_path), object_key=object_key)
        public_url = self.oss.build_public_url(object_key)

        return StorageUploadResult(
            local_path=str(local_path),
            public_url=public_url,
            storage_provider='oss',
            object_key=object_key,
        )

    def resolve_public_url(self, value: Optional[str]) -> Optional[str]:
        """解析照片 URL（支持本地路径转换为绝对 URL）"""
        # 处理：http/https URL
        # 处理：相对路径
        # 处理：/uploads/ 路径（转换为 absolute URL）

    def resolve_client_url(self, value: Optional[str]) -> Optional[str]:
        """
        解析客户端访问的 URL

        对于本地文件存储，绝对 tunnel/CDN URL 可能过期。
        如果路径指向 `/uploads/...`，返回相对路径，以便客户端使用当前 API origin。
        """
        resolved = self.resolve_public_url(value)
        if not resolved:
            return None

        parsed = urlparse(resolved)
        if parsed.scheme in {"http", "https"} and parsed.path.startswith('/uploads/'):
            # 保留查询字符串和片段
            suffix = ''
            if parsed.query:
                suffix += f'?{parsed.query}'
            if parsed.fragment:
                suffix += f'#{parsed.fragment}'
            return f'{parsed.path}{suffix}'

        return resolved
```

**URL 解析策略**:

| 场景 | 输入 | 输出 |
|------|------|------|
| OSS 存储已启用 | `/uploads/photos/abc.jpg` | `https://bucket.endpoint.com/photos/abc.jpg` |
| 本地存储 | `/uploads/photos/abc.jpg` | `http://api-origin/uploads/photos/abc.jpg` |
| HTTPS 远程 URL | `https://cdn.example.com/abc.jpg` | `https://cdn.example.com/abc.jpg` |
| 相对路径 | `/uploads/abc.jpg` | `http://api-origin/uploads/abc.jpg` |

### 7. 事件 AI 服务

**文件**: `backend/app/services/event_ai_service.py`

**功能**: 调用 AI 服务生成单个事件的故事

```python
def generate_event_story_for_event(
    db: Session,
    user_id: str,
    event_id: str,
    *,
    strict_ai: bool = True,
) -> Tuple[bool, Optional[str]]:
    """为单个事件生成故事"""

    # 1. 查询事件和照片
    event = db.scalar(select(Event).where(and_(Event.id == event_id, Event.user_id == user_id)))
    photos = list(db.scalars(
        select(Photo)
        .where(and_(Photo.user_id == user_id, Photo.event_id == event_id))
        .order_by(Photo.shoot_time.asc().nullslast(), Photo.created_at.asc())
    ).all())

    # 2. 检查 AI 配置
    if not ai_service.is_configured():
        if strict_ai:
            event.status = "ai_failed"
            event.ai_error = ai_service.configuration_error_code()
            event.title = ensure_event_title(event)
            db.commit()
        return False, ai_service.configuration_error_code()

    # 3. 解析公共 URL（AI 需要访问照片）
    public_urls: list[str] = []
    for photo in photos:
        resolved = storage_service.resolve_public_url(photo.thumbnail_url)
        if resolved:
            public_urls.append(resolved)

    if not public_urls:
        event.status = "ai_failed"
        event.ai_error = "photos_are_not_publicly_accessible_for_ai"
        event.title = ensure_event_title(event)
        db.commit()
        return False, event.ai_error

    # 4. 生成地理编码（如果缺失）
    location = get_event_location_text(event) or "未知地点"

    # 5. 分析照片（生成描述）
    analysis = ai_service.analyze_event_photos(
        event_id=event_id,
        photo_urls=public_urls,
        location=location,
    )
    descriptions = [str(d) for d in analysis.get("descriptions", []) if d]

    # 6. 生成故事
    story = ai_service.generate_event_story(
        event_id=event_id,
        location=location,
        start_time=event.start_time.isoformat(),
        end_time=event.end_time.isoformat(),
        photo_descriptions=descriptions,
    )

    # 7. 更新事件
    if not story:
        event.status = "ai_failed"
        event.ai_error = ai_service.last_error_code or "story_generation_failed"
        event.title = ensure_event_title(event)
        db.commit()
        return False, event.ai_error

    event.title = str(story.get("title") or ensure_event_title(event))
    event.story_text = str(story.get("story") or "").strip() or None

    emotion = story.get("emotion")
    if isinstance(emotion, str) and emotion:
        event.emotion_tag = emotion

    event.status = "generated"
    event.ai_error = None
    db.commit()
    return True, None
```

## ⚠️ 注意事项

### 1. 数据库会话管理

**问题**: 同一线程创建多个 Session 可能导致连接泄漏

**最佳实践**:
```python
# ✅ 推荐：依赖注入模式（FastAPI）
def get_db() -> Generator[Session, None, None]:
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

# API 路由使用
@router.post("/upload")
def upload_route(..., db: Session = Depends(get_db)):
    # 使用 db 会话
    db.commit()

# ❌ 避免：直接在服务中创建 Session
class PhotoService:
    def some_method(self, db: Session = SessionLocal()):  # 错误
        # ...
        db.close()
```

---

### 2. 缓存一致性

**问题**: 内存缓存（`SimpleCache`）不跨进程共享

**适用场景**:
- ✅ 单进程 FastAPI + Celery Worker（可以共享）
- ❌ 多进程 Worker（不共享缓存）

**解决方案**（未来）:
```python
# 使用 Redis 作为缓存后端（跨进程共享）
import redis

redis_client = redis.Redis(host='localhost', port=6379, decode_responses=True)

def redis_get(key: str):
    return redis_client.get(f"travel_album:{key}")

def redis_set(key: str, value: any, ttl: int):
    redis_client.setex(f"travel_album:{key}", ttl, value)
```

---

### 3. 异步任务超时

**当前配置**: `celery_task_time_limit = 3600`（1小时）

**风险**: AI 调用可能超时（网络延迟、模型响应慢）

**建议**:
- 监控任务执行时间
- 针对不同任务设置不同超时（聚类 10分钟，AI 生成 30分钟）
- 实现任务重试机制（Celery `autoretry_for`）

---

### 4. 存储策略

**当前**: 先保存本地，再上传到 OSS（如果启用）

**风险**:
- 本地磁盘空间不足
- OSS 上传失败时照片已保存但未在云端

**建议**:
- 添加 OSS 上传失败重试队列
- 实现临时 URL 签名（预签名 URL，避免永久公开）
- 定期清理本地旧文件

---

### 5. AI 配置切换

**当前**: 支持热重载（`force_reload=True`）

**建议**:
- 添加配置变更事件（通知正在执行的任务使用旧配置）
