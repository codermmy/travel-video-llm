## 1. 阶段概述

### 1.1 目标

**搭建完整的项目骨架**，包括：

1. ✅ 创建前端完整目录结构
2. ✅ 创建后端完整目录结构
3. ✅ 为每个模块创建占位文件/骨架代码
4. ✅ 配置开发工具链（代码规范、类型检查）
5. ✅ 验证前后端通信基础打通
6. ✅ 数据库初始化完成

**核心原则**：**先搭骨架，再填肉**。目录结构一旦确定，后续不再随意改动。

### 1.2 成功标准

- [ ] 前端项目目录完整，所有占位文件已创建
- [ ] 后端项目目录完整，所有占位文件已创建
- [ ] 前端可启动，显示欢迎页面
- [ ] 后端可启动，API 文档可访问
- [ ] 数据库连接正常，表结构创建完成
- [ ] 前后端通信测试通过（health 接口）

### 3.6 创建 API 服务层骨架

**services/api/client.ts**:

```typescript
import axios, { AxiosError, AxiosInstance } from 'axios';
import AsyncStorage from '@react-native-async-storage/async-storage';

// API 基础 URL
const API_BASE_URL = __DEV__
  ? 'http://localhost:8000/api/v1'
  : 'https://api.example.com/api/v1';

/**
 * 创建 Axios 实例
 */
export const apiClient: AxiosInstance = axios.create({
  baseURL: API_BASE_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

/**
 * 请求拦截器：自动添加 Token
 */
apiClient.interceptors.request.use(
  async (config) => {
    const token = await AsyncStorage.getItem('token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

/**
 * 响应拦截器：统一错误处理
 */
apiClient.interceptors.response.use(
  (response) => response.data,
  (error: AxiosError<ApiErrorResponse>) => {
    // TODO: 完善错误处理逻辑
    console.error('API Error:', error);
    return Promise.reject(error);
  }
);
```

```typescript
  token: null,
  isAuthenticated: false,
  isLoading: false,

  register: async (deviceId: string, nickname?: string) => {

    try {
      const response = await authService.register({ deviceId, nickname });
      await authService.saveToken(response.token);

      set({
        user: {
          userId: response.userId,
          deviceId: response.deviceId,
          nickname: response.nickname,
          createdAt: new Date().toISOString(),
        },
        token: response.token,
        isAuthenticated: true,
        isLoading: false,
      });

      set({ isLoading: false });
      throw error;
    }
  },

  logout: async () => {
    await authService.clearToken();
    set({
      user: null,
      token: null,
      isAuthenticated: false,
    });
  },

    const token = await authService.getToken();
    if (token) {
      set({ isAuthenticated: true, token });

    }
  },
```

```python
"""
认证和安全相关
"""
from datetime import datetime, timedelta
from typing import Optional
from jose import JWTError, jwt
from passlib.context import CryptContext
from app.config import get_settings

settings = get_settings()

# 密码加密上下文（虽然本项目不使用密码，但预留）
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def create_access_token(user_id: int, expires_delta: Optional[timedelta] = None) -> str:
    """
    创建 JWT Token

    Args:
        user_id: 用户 ID
        expires_delta: 过期时间增量

    Returns:
        JWT Token 字符串
    """
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(
            minutes=settings.JWT_ACCESS_TOKEN_EXPIRE_MINUTES
        )

    to_encode = {"user_id": user_id, "exp": expire}
    encoded_jwt = jwt.encode(
        to_encode, settings.JWT_SECRET_KEY, algorithm=settings.JWT_ALGORITHM
    )
    return encoded_jwt


def verify_token(token: str) -> Optional[int]:
    """
    验证 JWT Token

    Args:
        token: JWT Token 字符串

    Returns:
        用户 ID，验证失败返回 None
    """
    try:
        payload = jwt.decode(token, settings.JWT_SECRET_KEY, algorithms=[settings.JWT_ALGORITHM])
        user_id: int = payload.get("user_id")
        return user_id
    except JWTError:
        return None
```

```python
"""
自定义异常和异常处理
"""
from fastapi import Request, HTTPException
from fastapi.responses import JSONResponse
from loguru import logger


class AppException(Exception):
    """应用基础异常"""

    def __init__(self, message: str, code: str = "APP_ERROR"):
        self.message = message
        self.code = code
        super().__init__(message)


class PhotoNotFoundException(AppException):
    """照片不存在异常"""

    def __init__(self, photo_id: int):
        super().__init__(
            message=f"照片 {photo_id} 不存在",
            code="PHOTO_NOT_FOUND",
        )


class EventNotFoundException(AppException):
    """事件不存在异常"""

    def __init__(self, event_id: int):
        super().__init__(
            message=f"事件 {event_id} 不存在",
            code="EVENT_NOT_FOUND",
        )


class UserNotFoundException(AppException):
    """用户不存在异常"""

    def __init__(self, user_id: int):
        super().__init__(
            message=f"用户 {user_id} 不存在",
            code="USER_NOT_FOUND",
        )


async def app_exception_handler(request: Request, exc: AppException):
    """应用异常处理器"""
    logger.error(f"App exception: {exc}")
    return JSONResponse(
        status_code=400,
        content={
            "success": False,
            "error": {
                "code": exc.code,
                "message": exc.message,
            },
            "timestamp": None,
        },
    )


async def http_exception_handler(request: Request, exc: HTTPException):
    """HTTP 异常处理器"""
    return JSONResponse(
        status_code=exc.status_code,
        content={
            "success": False,
            "error": {
                "code": f"HTTP_{exc.status_code}",
                "message": exc.detail,
            },
            "timestamp": None,
        },
    )


async def general_exception_handler(request: Request, exc: Exception):
    """通用异常处理器"""
    logger.error(f"Unhandled exception: {exc}")
    return JSONResponse(
        status_code=500,
        content={
            "success": False,
            "error": {
                "code": "INTERNAL_ERROR",
                "message": "服务器内部错误",
            },
            "timestamp": None,
        },
    )
```

```python
"""
认证接口
"""
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from app.db.session import get_db
from app.schemas.user import UserRegister, AuthResponse
from app.core.security import create_access_token
from app.models.user import User

router = APIRouter()


@router.post("/register", response_model=AuthResponse)
async def register(
    user_data: UserRegister,
    db: Session = Depends(get_db)
):
    """
    设备注册/登录

    TODO: 实现完整的注册逻辑
    """
    # 检查设备是否已存在
    existing_user = db.query(User).filter(User.device_id == user_data.device_id).first()

    if existing_user:
        # 返回现有用户
        token = create_access_token(existing_user.id)
        return AuthResponse(
            userId=existing_user.id,
            deviceId=existing_user.device_id,
            nickname=existing_user.nickname or "旅行者",
            token=token,
            isNewUser=False,
        )

    # 创建新用户
    new_user = User(
        device_id=user_data.device_id,
        nickname=user_data.nickname or "旅行者",
    )
    db.add(new_user)
    db.commit()
    db.refresh(new_user)

    token = create_access_token(new_user.id)

    return AuthResponse(
        userId=new_user.id,
        deviceId=new_user.device_id,
        nickname=new_user.nickname,
        token=token,
        isNewUser=True,
    )
```
