```python
"""
用户相关 Schema
"""
from pydantic import BaseModel, Field
from datetime import datetime


class UserRegister(BaseModel):
    """用户注册请求"""
    device_id: str = Field(..., min_length=1, max_length=100, description="设备唯一标识")
    nickname: str | None = Field(None, max_length=50, description="用户昵称")


class UserResponse(BaseModel):
    """用户响应"""
    userId: int
    deviceId: str
    nickname: str
    createdAt: datetime

    class Config:
        from_attributes = True


class AuthResponse(BaseModel):
    """认证响应"""
    userId: int
    deviceId: str
    nickname: str
    token: str
    isNewUser: bool = False
    createdAt: datetime
```

### 3.5 完整代码实现

#### `app/api/v1/auth.py`

```python
"""
认证接口
"""
from fastapi import APIRouter, HTTPException, status
from sqlalchemy.orm import Session
from datetime import datetime

from app.models.base import get_db
from app.models.user import User
from app.schemas.user import UserRegister, AuthResponse
from app.schemas.common import ApiResponse
from app.core.security import create_access_token

router = APIRouter()


@router.post("/register", response_model=ApiResponse[AuthResponse])
async def register(
    user_data: UserRegister,
    db: Session = Depends(get_db)
):
    """
    设备注册/登录

    根据 device_id 查询或创建用户，返回 JWT Token。

    Args:
        user_data: 注册请求数据
        db: 数据库会话

    Returns:
        认证响应，包含用户信息和 Token

    Raises:
        HTTPException(500): 数据库操作失败
    """
    try:
        # 1. 查询设备是否已注册
        existing_user = db.query(User).filter(
            User.device_id == user_data.device_id
        ).first()

        if existing_user:
            # 老用户：直接返回 Token
            token = create_access_token(existing_user.id)
            return ApiResponse(
                success=True,
                data=AuthResponse(
                    userId=existing_user.id,
                    deviceId=existing_user.device_id,
                    nickname=existing_user.nickname or "旅行者",
                    token=token,
                    isNewUser=False,
                    createdAt=existing_user.created_at
                ),
                timestamp=datetime.utcnow()
            )

        # 2. 新用户：创建记录
        nickname = user_data.nickname or "旅行者"

        new_user = User(
            device_id=user_data.device_id,
            nickname=nickname
        )
        db.add(new_user)
        db.commit()
        db.refresh(new_user)

        # 3. 生成 Token
        token = create_access_token(new_user.id)

        return ApiResponse(
            success=True,
            data=AuthResponse(
                userId=new_user.id,
                deviceId=new_user.device_id,
                nickname=new_user.nickname,
                token=token,
                isNewUser=True,
                createdAt=new_user.created_at
            ),
            timestamp=datetime.utcnow()
        )

    except Exception as e:
        db.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"注册失败: {str(e)}"
        )


# 可选：获取当前用户信息
@router.get("/me", response_model=ApiResponse[UserResponse])
```

## 4. 预期行为

### 4.1 新用户注册流程

```
1. 前端生成 UUID 作为 device_id
2. 调用 POST /api/v1/auth/register
3. 后端创建用户记录
4. 返回 Token + isNewUser=true
5. 前端存储 Token 到 AsyncStorage
```

### 4.2 老用户登录流程

```
1. 前端从 AsyncStorage 读取 device_id
2. 调用 POST /api/v1/auth/register
3. 后端查询到已有用户
4. 返回 Token + isNewUser=false
5. 前端更新 Token
```

### 4.3 错误场景

| 场景 | HTTP状态码 | 响应 |
|------|-----------|------|
| device_id 为空 | 422 | 验证错误 |
| device_id 超长 | 422 | 验证错误 |
| 数据库错误 | 500 | "注册失败" |

```python
from fastapi.testclient import TestClient

client = TestClient(app)


def test_register_new_user(db: Session):

        "/api/v1/auth/register",

    data = response.json()
    assert data["success"] is True
    assert data["data"]["isNewUser"] is True

    assert "token" in data["data"]

    assert user.nickname == "测试用户"
```

## 6. 风险与注意事项

### 6.1 数据完整性

| 风险 | 应对措施 |
|------|----------|
| device_id 重复 | 数据库 UNIQUE 约束保证 |
| 昵称为空 | 设置默认值 "旅行者" |

### 6.2 安全考虑

1. **device_id 唯一性**：前端使用 UUID 生成，确保全局唯一
2. **Token 安全**：30天有效期是简化设计，生产环境建议缩短
3. **设备数限制**：当前版本不限制单用户设备数
