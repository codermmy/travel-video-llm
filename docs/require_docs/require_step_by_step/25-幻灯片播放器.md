## 2. 实现细节

### 2.1 幻灯片播放器组件

```typescript
/**
 * 幻灯片播放器
 */
import React, { useRef, useState, useEffect } from 'react';
import { View, Text, StyleSheet, Pressable, Image } from 'react-native';
import { ViewPager } from 'react-native-pager-view';
import { Audio } from 'expo-av';

interface SlideshowProps {
  photos: Photo[];
  musicUrl?: string;
  onClose: () => void;
}

export function SlideshowPlayer({ photos, musicUrl, onClose }: SlideshowProps) {
  const [currentIndex, setCurrentIndex] = useState(0);
  const [isPlaying, setIsPlaying] = useState(true);
  const pagerRef = useRef<ViewPager>(null);
  const soundRef = useRef(new Audio.Sound());

  useEffect(() => {
    // 自动播放
    const interval = setInterval(() => {
      if (isPlaying && currentIndex < photos.length - 1) {
        pagerRef.current?.setPage(currentIndex + 1);
      }
    }, 5000);

    return () => clearInterval(interval);
  }, [isPlaying, currentIndex]);

  return (
    <View style={styles.container}>
      {/* 顶部控制栏 */}
      <View style={styles.header}>
        <Pressable onPress={onClose}>
          <Text style={styles.closeButton}>✕</Text>
        </Pressable>
        <Text style={styles.counter}>
          {currentIndex + 1} / {photos.length}
        </Text>
      </View>

      {/* 照片 */}
      <ViewPager
        ref={pagerRef}
        style={styles.pager}
        initialPage={0}
        onPageSelected={(index) => setCurrentIndex(index)}
      >
        {photos.map((photo) => (
          <View key={photo.id} style={styles.slide}>
            <Image source={{ uri: photo.thumbnailUrl }} style={styles.image} />
            {photo.storyText && (
              <View style={styles.caption}>
                <Text style={styles.captionText}>{photo.storyText}</Text>
              </View>
            )}
          </View>
        ))}
      </ViewPager>

      {/* 底部控制 */}
      <View style={styles.controls}>
        <Pressable onPress={() => pagerRef.current?.setPage(currentIndex - 1)}>
          <Text style={styles.controlButton}>◀</Text>
        </Pressable>
        <Pressable onPress={() => setIsPlaying(!isPlaying)}>
          <Text style={styles.controlButton}>{isPlaying ? '⏸' : '▶'}</Text>
        </Pressable>
        <Pressable onPress={() => pagerRef.current?.setPage(currentIndex + 1)}>
          <Text style={styles.controlButton}>▶</Text>
        </Pressable>
      </View>
    </View>
  );
}
```
