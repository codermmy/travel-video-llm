## 2. 实现细节

### 2.1 照片网格组件

```typescript
/**
 * 照片网格组件
 */
import React from 'react';
import { View, StyleSheet, FlatList, Image, Dimensions, Pressable } from 'react-native';

const { width } = Dimensions.get('window');
const COLUMN_WIDTH = (width - 32) / 3;

interface PhotoGridProps {
  photos: Photo[];
  onPhotoPress: (photo: Photo) => void;
  numColumns?: number;
}

export function PhotoGrid({ photos, onPhotoPress, numColumns = 3 }: PhotoGridProps) {
  return (
    <FlatList
      data={photos}
      keyExtractor={(item) => item.id}
      numColumns={numColumns}
      renderItem={({ item }) => (
        <Pressable onPress={() => onPhotoPress(item)}>
          <Image source={{ uri: item.thumbnailUrl }} style={styles.photo} />
        </Pressable>
      )}
      contentContainerStyle={styles.gridContent}
    />
  );
}

const styles = StyleSheet.create({
  gridContent: { gap: 4 },
  photo: { width: COLUMN_WIDTH, height: COLUMN_WIDTH, borderRadius: 4 },
});
```

### 2.2 照片查看器

```typescript
/**
 * 照片查看器
 */
import React, { useState } from 'react';
import { View, Image, StyleSheet, Pressable, Text } from 'react-native';
import { ViewPager } from 'react-native-pager-view';

interface PhotoViewerProps {
  photos: Photo[];
  initialIndex: number;
  onClose: () => void;
}

export function PhotoViewer({ photos, initialIndex, onClose }: PhotoViewerProps) {
  const [currentIndex, setCurrentIndex] = useState(initialIndex);

  const currentPhoto = photos[currentIndex];

  return (
    <View style={styles.container}>
      {/* 顶部栏 */}
      <View style={styles.header}>
        <Pressable onPress={onClose}>
          <Text style={styles.backButton}>← 返回</Text>
        </Pressable>
        <Text style={styles.indexText}>
          {currentIndex + 1} / {photos.length}
        </Text>
      </View>

      {/* 照片 */}
      <ViewPager
        style={styles.pager}
        initialPage={initialIndex}
        onPageSelected={(index) => setCurrentIndex(index)}
      >
        {photos.map((photo) => (
          <View key={photo.id} style={styles.photoContainer}>
            <Image
              source={{ uri: photo.thumbnailUrl }}
              style={styles.photo}
              resizeMode="contain"
            />
          </View>
        ))}
      </ViewPager>

      {/* 底部信息 */}
      <View style={styles.footer}>
        {currentPhoto.shootTime && (
          <Text style={styles.info}>
            📅 {formatDateTime(currentPhoto.shootTime)}
          </Text>
        )}
        {currentPhoto.gpsLat && (
          <Text style={styles.info}>
            📍 {currentPhoto.gpsLat.toFixed(4)}, {currentPhoto.gpsLon.toFixed(4)}
          </Text>
        )}
      </View>
    </View>
  );
}
```
