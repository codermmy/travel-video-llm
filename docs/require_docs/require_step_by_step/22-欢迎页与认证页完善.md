/**
 * 欢迎页/注册页
 */
import { useState } from 'react';
import { View, Text, TextInput, StyleSheet, Pressable, ActivityIndicator } from 'react-native';
import { useRouter } from 'expo-router';
import { useAuthStore } from '@/stores';
import { requestPhotoLibraryPermission } from '@/utils/permissionUtils';

export default function WelcomeScreen() {
  const router = useRouter();
  const { register, isLoading, error, clearError } = useAuthStore();
  const [nickname, setNickname] = useState('');
  const [permissionDenied, setPermissionDenied] = useState(false);

  const handleGetStarted = async () => {
    clearError();

    // 1. 请求相册权限
    const { granted, canAskAgain } = await requestPhotoLibraryPermission();

    if (!granted) {
      if (canAskAgain) {
        setPermissionDenied(true);
      }
      return;
    }

    // 2. 注册
    const success = await register(nickname || undefined);

    if (success) {
      router.replace('/(tabs)');
    }
  };

  const handleSettings = () => {
    // 引导用户去设置
    // TODO: 实现跳转到系统设置
  };

  return (
    <View style={styles.container}>
      <View style={styles.content}>
        {/* Logo */}
        <View style={styles.logoContainer}>
          <Text style={styles.logo}>📸</Text>
        </View>

        {/* 标题 */}
        <Text style={styles.title}>旅行相册智能整理</Text>
        <Text style={styles.subtitle}>Your Travel Memories, Organized</Text>

        {/* 输入昵称 */}
        <TextInput
          style={styles.input}
          placeholder="输入昵称（可选）"
          value={nickname}
          onChangeText={setNickname}
          maxLength={20}
        />

        {/* 错误提示 */}
        {error ? <Text style={styles.error}>{error}</Text> : null}

        {/* 权限被拒提示 */}
        {permissionDenied && (
          <View style={styles.permissionBox}>
            <Text style={styles.permissionText}>
              需要相册权限才能整理照片
            </Text>
            <Pressable onPress={handleSettings}>
              <Text style={styles.settingsLink}>去设置开启</Text>
            </Pressable>
          </View>
        )}

        {/* 开始按钮 */}
        <Pressable
          style={({ pressed }) => [
            styles.button,
            pressed && styles.buttonPressed,
            isLoading && styles.buttonDisabled,
          ]}
          onPress={handleGetStarted}
          disabled={isLoading}
        >
          {isLoading ? (
            <ActivityIndicator color="#fff" />
          ) : (
            <Text style={styles.buttonText}>开始使用</Text>
          )}
        </Pressable>

        <Text style={styles.terms}>
          开始使用即表示您同意我们的服务条款
        </Text>
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#f5f5f5' },
  content: { flex: 1, justifyContent: 'center', alignItems: 'center', padding: 24 },
  logoContainer: { marginBottom: 24 },
  logo: { fontSize: 64 },
  title: { fontSize: 28, fontWeight: 'bold', marginBottom: 8, color: '#333' },
  subtitle: { fontSize: 16, color: '#666', marginBottom: 48 },
  input: {
    width: '100%', height: 48, backgroundColor: '#fff',
    borderRadius: 8, paddingHorizontal: 16, marginBottom: 16,
    borderWidth: 1, borderColor: '#ddd',
  },
  error: { color: '#f44336', marginBottom: 16, fontSize: 14 },
  permissionBox: {
    backgroundColor: '#fff3e0', padding: 16, borderRadius: 8,
    marginBottom: 16, width: '100%', alignItems: 'center',
  },
  permissionText: { color: '#b71c1c', marginBottom: 8 },
  settingsLink: { color: '#6200EE' },
  button: {
    width: '100%', height: 48, backgroundColor: '#6200EE',
    borderRadius: 8, justifyContent: 'center', alignItems: 'center',
    marginBottom: 16,
  },
  buttonPressed: { opacity: 0.8 },
  buttonDisabled: { opacity: 0.6 },
  buttonText: { color: '#fff', fontSize: 16, fontWeight: '600' },
  terms: { fontSize: 12, color: '#999', textAlign: 'center' },
});
