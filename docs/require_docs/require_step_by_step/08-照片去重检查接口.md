- 查询数据库返回已存在的哈希

```python
"""
照片相关 Schema
"""
from pydantic import BaseModel, Field
from typing import List


class CheckDuplicatesRequest(BaseModel):
    """去重检查请求"""
    hashes: List[str] = Field(..., min_length=1, max_length=1000, description="哈希列表")


class CheckDuplicatesResponse(BaseModel):
    """去重检查响应"""
    newHashes: List[str] = Field(default_factory=list, description="需要上传的哈希")
    existingHashes: List[str] = Field(default_factory=list, description="已存在的哈希")
    totalCount: int = Field(..., description="总数量")
```

### 3.4 核心代码实现

#### `app/api/v1/photos.py`

```python
"""
照片接口
"""
from typing import List
from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session
from sqlalchemy import select, or_

from app.models.base import get_db
from app.models.photo import Photo
from app.schemas.photo import CheckDuplicatesRequest, CheckDuplicatesResponse
from app.schemas.common import ApiResponse
from app.api.deps import CurrentUserIdDep
from datetime import datetime

router = APIRouter()


@router.post("/check-duplicates", response_model=ApiResponse[CheckDuplicatesResponse])
async def check_duplicates(
    request: CheckDuplicatesRequest,
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """
    检查照片哈希是否已存在

    业务逻辑：
    1. 接收前端发送的哈希列表
    2. 查询数据库中该用户已有的照片哈希
    3. 返回需要上传的新照片哈希

    Args:
        request: 包含哈希列表的请求体
        user_id: 当前用户 ID（从 Token 解析）
        db: 数据库会话

    Returns:
        去重检查结果
    """
    # 防止请求过大
    if len(request.hashes) > 1000:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="哈希数量不能超过 1000 个"
        )

    try:
        # 1. 查询该用户已有的照片哈希
        # 使用 file_hash IN (...) 查询
        existing_photos = db.query(Photo.file_hash).filter(
            Photo.user_id == user_id,
            Photo.file_hash.in_(request.hashes)
        ).all()

        # 提取已存在的哈希集合
        existing_hashes = {photo.file_hash for photo in existing_photos}

        # 2. 分类哈希
        new_hashes = [h for h in request.hashes if h not in existing_hashes]

        return ApiResponse(
            success=True,
            data=CheckDuplicatesResponse(
                newHashes=new_hashes,
                existingHashes=list(existing_hashes),
                totalCount=len(request.hashes)
            ),
            timestamp=datetime.utcnow()
        )

    except Exception as e:
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"去重检查失败: {str(e)}"
        )


@router.get("/")
async def get_photos(
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """
    获取用户所有照片

    TODO: 实现分页、过滤等功能
    """
    photos = db.query(Photo).filter(Photo.user_id == user_id).all()

    return {
        "success": True,
        "data": {
            "photos": [
                {
                    "id": p.id,
                    "fileHash": p.file_hash,
                    "thumbnailUrl": p.thumbnail_url,
                    "gpsLat": float(p.gps_lat) if p.gps_lat else None,
                    "gpsLon": float(p.gps_lon) if p.gps_lon else None,
                    "shootTime": p.shoot_time.isoformat() if p.shoot_time else None,
                    "eventId": p.event_id,
                }
                for p in photos
            ]
        },
        "timestamp": datetime.utcnow().isoformat()
    }


@router.get("/{photo_id}")
async def get_photo(
    photo_id: int,
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """
    获取单张照片详情
    """
    photo = db.query(Photo).filter(
        Photo.id == photo_id,
        Photo.user_id == user_id
    ).first()

    if not photo:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="照片不存在"
        )

    return {
        "success": True,
        "data": {
            "id": photo.id,
            "fileHash": photo.file_hash,
            "thumbnailUrl": photo.thumbnail_url,
            "gpsLat": float(photo.gps_lat) if photo.gps_lat else None,
            "gpsLon": float(photo.gps_lon) if photo.gps_lon else None,
            "shootTime": photo.shoot_time.isoformat() if photo.shoot_time else None,
            "eventId": photo.event_id,
        },
        "timestamp": datetime.utcnow().isoformat()
    }
```

```python
def test_check_duplicates_all_new(db, test_user, client, auth_headers):
    """测试全部是新照片"""
    hashes = ["hash001", "hash002", "hash003"]

    response = client.post(
        "/api/v1/photos/check-duplicates",
        json={"hashes": hashes},
        headers=auth_headers
    )

    assert response.status_code == 200
    data = response.json()["data"]
    assert len(data["newHashes"]) == 3
    assert len(data["existingHashes"]) == 0


def test_check_duplicates_some_exist(db, test_user, client, auth_headers):
    """测试部分已存在"""
    # 先插入一张照片
    photo = Photo(
        user_id=test_user.id,
        file_hash="hash001",
        thumbnail_path="/test/path.jpg"
    )
    db.add(photo)
    db.commit()

    hashes = ["hash001", "hash002", "hash003"]
    response = client.post(
        "/api/v1/photos/check-duplicates",
        json={"hashes": hashes},
        headers=auth_headers
    )

    assert response.status_code == 200
    data = response.json()["data"]
    assert set(data["existingHashes"]) == {"hash001"}
    assert set(data["newHashes"]) == {"hash002", "hash003"}
```
