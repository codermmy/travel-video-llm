"""
照片相关 Schema
"""
from pydantic import BaseModel, Field
from typing import List, Optional
from datetime import datetime
from decimal import Decimal


class PhotoUploadItem(BaseModel):
    """单张照片上传数据"""
    hash: str = Field(..., description="SHA-256 哈希值")
    gpsLat: Optional[Decimal] = Field(None, description="纬度")
    gpsLon: Optional[Decimal] = Field(None, description="经度")
    shootTime: Optional[datetime] = Field(None, description="拍摄时间")


class PhotoUploadRequest(BaseModel):
    """批量照片上传请求（元数据部分）"""
    photos: List[PhotoUploadItem] = Field(..., min_length=1, max_length=50)


class PhotoUploadResponse(BaseModel):
    """照片上传响应"""
    uploaded: int = Field(..., description="成功上传数量")
    failed: int = Field(..., description="失败数量")
    taskId: Optional[str] = Field(None, description="聚类任务 ID")
```

"""
文件存储服务
"""
import os
import uuid
from pathlib import Path
from fastapi import UploadFile
from werkzeug.utils import secure_filename

from app.config import get_settings

settings = get_settings()


class StorageService:
    """本地文件存储服务"""

    def __init__(self):
        self.upload_dir = Path(settings.UPLOAD_DIR) / "photos"
        self.upload_dir.mkdir(parents=True, exist_ok=True)

    def get_photo_path(self, photo_id: int, extension: str = "jpg") -> str:
        """获取照片存储路径"""
        filename = f"{photo_id}.{extension}"
        return str(self.upload_dir / filename)

    async def save_photo(
        self,
        photo_id: int,
        file: UploadFile,
        filename: str | None = None
    ) -> tuple[str, int]:
        """
        保存照片文件

        Args:
            photo_id: 照片 ID
            file: 上传的文件
            filename: 可选的自定义文件名

        Returns:
            (文件路径, 文件大小)

        Raises:
            Exception: 保存失败
        """
        # 确定文件扩展名
        if filename:
            ext = Path(filename).suffix or ".jpg"
        else:
            ext = Path(file.filename or "photo.jpg").suffix or ".jpg"

        # 目标路径
        target_path = self.get_photo_path(photo_id, ext.lstrip("."))

        # 保存文件
        try:
            contents = await file.read()

            with open(target_path, "wb") as f:
                f.write(contents)

            file_size = len(contents)

            return target_path, file_size

        except Exception as e:
            raise Exception(f"保存文件失败: {str(e)}")

    async def delete_photo(self, photo_id: int, extension: str = "jpg") -> bool:
        """
        删除照片文件

        Args:
            photo_id: 照片 ID
            extension: 文件扩展名

        Returns:
            是否删除成功
        """
        target_path = self.get_photo_path(photo_id, extension)

        try:
            if os.path.exists(target_path):
                os.remove(target_path)
                return True
            return False
        except Exception:
            return False

    def get_photo_url(self, photo_id: int, extension: str = "jpg") -> str:
        """
        获取照片访问 URL

        当前版本返回本地路径，生产环境应返回 OSS URL
        """
        return self.get_photo_path(photo_id, extension)


# 导出单例

async def upload_photos_metadata(
    request: dict,
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """
    上传照片元数据（简化版接口）

    前端先上传文件到临时目录，然后调用此接口保存元数据

    请求体：
    {
      "photos": [
        {
          "hash": "xxx",
          "thumbnailPath": "/path/to/thumbnail",
          "gpsLat": 30.259,
          "gpsLon": 120.215,
          "shootTime": "2024-10-01T10:30:00Z"
        }
      ]
    }
    """
    from decimal import Decimal

    photos_data = request.get("photos", [])

    if not photos_data:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="没有照片数据"
        )

    uploaded = 0
    failed = 0

    for photo_data in photos_data:
        try:
            # 检查哈希是否已存在
            existing = db.query(Photo).filter(
                Photo.user_id == user_id,
                Photo.file_hash == photo_data["hash"]
            ).first()

            if existing:
                failed += 1
                continue

            # 创建照片记录
            photo = Photo(
                user_id=user_id,
                file_hash=photo_data["hash"],
                local_path=photo_data.get("thumbnailPath", ""),
                thumbnail_path=photo_data.get("thumbnailPath", ""),
                thumbnail_url=photo_data.get("thumbnailPath", ""),
                gps_lat=Decimal(str(photo_data["gpsLat"])) if photo_data.get("gpsLat") else None,
                gps_lon=Decimal(str(photo_data["gpsLon"])) if photo_data.get("gpsLon") else None,
                shoot_time=datetime.fromisoformat(photo_data["shootTime"]) if photo_data.get("shootTime") else None,
                status="uploaded"
            )

            db.add(photo)
            uploaded += 1

        except Exception as e:
            print(f"保存照片失败: {e}")
            failed += 1

    try:
        db.commit()

        # TODO: 触发聚类任务

        if uploaded > 0:

        return ApiResponse(
            success=True,
            data=PhotoUploadResponse(
                uploaded=uploaded,
                failed=failed,
                taskId=task_id
            ),
            timestamp=datetime.utcnow()
        )

    except Exception as e:
        db.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,

/**
 * 照片上传相关 API
 */
import { apiClient } from './client';
import type { PhotoMetadata } from '@/types';

interface UploadPhotoRequest {
  hash: string;
  thumbnailPath: string;
  gpsLat?: number;
  gpsLon?: number;
  shootTime: string;
}

interface UploadPhotosResponse {
  uploaded: number;
  failed: number;
  taskId?: string;
}

export const photoUploadService = {
  /**
   * 检查照片是否已存在（去重）
   */
  async checkDuplicates(hashes: string[]): Promise<{
    newHashes: string[];
    existingHashes: string[];
  }> {
    const response = await apiClient<{
      success: true;
      data: {
        newHashes: string[];
        existingHashes: string[];
        totalCount: number;
      };
      timestamp: string;
    }>({
      method: 'POST',
      url: '/photos/check-duplicates',
      data: { hashes },
    });

    return {
      newHashes: response.data.newHashes,
      existingHashes: response.data.existingHashes,
    };
  },

  /**
   * 上传照片元数据
   */
  async uploadPhotos(
    photos: UploadPhotoRequest[]
  ): Promise<UploadPhotosResponse> {
    const response = await apiClient<{
      success: true;
      data: UploadPhotosResponse;
      timestamp: string;
    }>({
      method: 'POST',
      url: '/photos/upload/metadata',
      data: { photos },
    });

    return response.data;
  },

  /**
   * 获取用户照片列表
   */
  async getPhotos(): Promise<any[]> {
    const response = await apiClient<{
      success: true;
      data: { photos: any[] };
      timestamp: string;
    }>({
      method: 'GET',
      url: '/photos',
    });

    return response.data.photos;
  },
};

## 4. 预期行为

### 4.1 上传流程

```
1. 前端扫描相册，计算哈希
2. 调用去重接口，获取需要上传的照片
3. 生成缩略图
4. 调用上传接口，保存元数据
5. 后端触发聚类任务
6. 返回上传结果
