### 3.2 聚类算法设计

#### 算法原理

基于 DBSCAN（Density-Based Spatial Clustering of Applications with Noise）改进的时空聚类算法：

**距离定义**：

```python
# 两张照片之间的"时空距离"
distance = sqrt(
    (spatial_distance / distance_threshold)^2 +
    (temporal_distance / time_threshold)^2
)
```

其中：
- `spatial_distance` = Haversine 球面距离（公里）
- `temporal_distance` = 时间差（小时）

#### 聚类参数

```python
TIME_THRESHOLD_HOURS = 48      # 时间阈值：48小时内
DISTANCE_THRESHOLD_KM = 50      # 距离阈值：50公里内
MIN_PHOTOS_PER_EVENT = 5        # 每个事件最少5张照片
```

#### 处理无 GPS 照片

**策略**：无 GPS 照片仅使用时间距离聚类

```python
if photo1.has_gps and photo2.has_gps:
    # 两者都有 GPS：使用时空距离
    distance = spacetime_distance(photo1, photo2)
else:
    # 至少一个无 GPS：只使用时间距离
    distance = temporal_distance(photo1, photo2)
```

```python
"""
地理计算工具
"""
import math
from typing import Tuple
from decimal import Decimal


def haversine_distance(
    lat1: float | Decimal,
    lon1: float | Decimal,
    lat2: float | Decimal,
    lon2: float | Decimal
) -> float:
    """
    计算两点之间的球面距离（Haversine 公式）

    Args:
        lat1, lon1: 第一个点的纬度、经度
        lat2, lon2: 第二个点的纬度、经度

    Returns:
        距离（公里）

    Example:
        >>> d = haversine_distance(30.259, 120.215, 30.267, 120.220)
        >>> print(f"{d:.2f} km")
        0.78 km
    """
    # 转换为弧度
    lat1, lon1, lat2, lon2 = map(float, [lat1, lon1, lat2, lon2])

    # Haversine 公式
    dlat = math.radians(lat2 - lat1)
    dlon = math.radians(lon2 - lon1)

    a = (
        math.sin(dlat / 2) ** 2 +
        math.cos(math.radians(lat1)) *
        math.cos(math.radians(lat2)) *
        math.sin(dlon / 2) ** 2
    )

    c = 2 * math.asin(math.sqrt(a))

    # 地球半径（公里）
    r = 6371

    return c * r


def calculate_center_point(
    points: list[Tuple[float, float]]
) -> Tuple[float, float]:
    """
    计算多个点的中心点（简单的平均值）

    Args:
        points: [(lat, lon), ...] 点的列表

    Returns:
        (平均纬度, 平均经度)
    """
    if not points:
        return (0.0, 0.0)

    avg_lat = sum(p[0] for p in points) / len(points)
    avg_lon = sum(p[1] for p in points) / len(points)

    return (avg_lat, avg_lon)


def format_coordinates(lat: float, lon: float) -> str:
    """格式化坐标为字符串"""
    return f"{abs(lat):.4f}°{'N' if lat >= 0 else 'S'}, {abs(lon):.4f}°{'E' if lon >= 0 else 'W'}"
```

#### `app/services/clustering_service.py`

```python
"""
时空聚类服务
"""
from datetime import datetime, timedelta
from typing import List, Tuple, Set, Optional
from sqlalchemy.orm import Session

from app.models.photo import Photo
from app.models.event import Event
from app.utils.geo import haversine_distance, calculate_center_point
import numpy as np
from sklearn.cluster import DBSCAN


# 聚类参数配置
class ClusteringConfig:
    TIME_THRESHOLD_HOURS = 48      # 时间阈值（小时）
    DISTANCE_THRESHOLD_KM = 50      # 距离阈值（公里）
    MIN_PHOTOS_PER_EVENT = 5        # 最少照片数

    @classmethod
    def from_env(cls):
        """从环境变量加载配置"""
        import os
        return cls(
            TIME_THRESHOLD_HOURS=int(os.getenv('CLUSTERING_TIME_THRESHOLD_HOURS', 48)),
            DISTANCE_THRESHOLD_KM=float(os.getenv('CLUSTERING_DISTANCE_THRESHOLD_KM', 50)),
            MIN_PHOTOS_PER_EVENT=int(os.getenv('CLUSTERING_MIN_PHOTOS', 5))
        )


class PhotoData:
    """用于聚类的照片数据"""

    def __init__(self, photo: Photo):
        self.id = photo.id
        self.user_id = photo.user_id
        self.file_hash = photo.file_hash
        self.has_gps = photo.gps_lat is not None and photo.gps_lon is not None
        self.lat = float(photo.gps_lat) if photo.gps_lat else None
        self.lon = float(photo.gps_lon) if photo.gps_lon else None
        self.shoot_time = photo.shoot_time

        # 用于聚类的时间戳（小时级，用于计算）
        if photo.shoot_time:
            self.timestamp = photo.shoot_time.timestamp()
        else:
            self.timestamp = datetime.utcnow().timestamp()


class SpacetimeClustering:
    """时空聚类算法"""

    def __init__(self, config: ClusteringConfig = None):
        self.config = config or ClusteringConfig()

    def prepare_photo_data(self, photos: List[Photo]) -> List[PhotoData]:
        """准备照片数据"""
        return [PhotoData(p) for p in photos]

    def calculate_distance_matrix(
        self,
        photo_data_list: List[PhotoData]
    ) -> np.ndarray:
        """
        计算照片之间的距离矩阵

        Returns:
            n x n 距离矩阵，其中 matrix[i][j] 是照片 i 和 j 的距离
        """
        n = len(photo_data_list)
        matrix = np.zeros((n, n))

        for i in range(n):
            for j in range(i + 1, n):
                distance = self._photo_distance(photo_data_list[i], photo_data_list[j])
                matrix[i][j] = distance
                matrix[j][i] = distance  # 对称矩阵

        return matrix

    def _photo_distance(self, p1: PhotoData, p2: PhotoData) -> float:
        """
        计算两张照片之间的时空距离

        Returns:
            归一化距离（0-1 之间，1 表示阈值距离）
        """
        # 1. 时间距离（小时）
        time_diff_hours = abs(p1.timestamp - p2.timestamp) / 3600

        # 2. 空间距离（如果有 GPS）
        if p1.has_gps and p2.has_gps:
            spatial_diff_km = haversine_distance(p1.lat, p1.lon, p2.lat, p2.lon)

            # 归一化并计算欧式距离
            normalized_time = min(time_diff_hours / self.config.TIME_THRESHOLD_HOURS, 1)
            normalized_spatial = min(spatial_diff_km / self.config.DISTANCE_THRESHOLD_KM, 1)

            return math.sqrt(normalized_time ** 2 + normalized_spatial ** 2)

        else:
            # 至少一个无 GPS，只用时间
            return min(time_diff_hours / self.config.TIME_THRESHOLD_HOURS, 1)

    def cluster(
        self,
        photo_data_list: List[PhotoData]
    ) -> List[List[int]]:
        """
        执行聚类算法

        Returns:
            聚类结果，每个子列表是一个簇的索引数组
            索引对应 photo_data_list 的下标
        """
        if len(photo_data_list) < self.config.MIN_PHOTOS_PER_EVENT:
            return []

        # 计算距离矩阵
        distance_matrix = self.calculate_distance_matrix(photo_data_list)

        # DBSCAN 聚类
        # eps 参数设置为 1.0（因为距离已归一化）
        # min_samples 对应最小照片数
        clustering = DBSCAN(
            eps=1.0,
            min_samples=max(2, self.config.MIN_PHOTOS_PER_EVENT // 2),
            metric='precomputed'
        )

        labels = clustering.fit_predict(distance_matrix)

        # 整理结果
        clusters = {}
        for idx, label in enumerate(labels):
            if label == -1:  # 噪声点
                continue
            if label not in clusters:
                clusters[label] = []
            clusters[label].append(idx)

        # 过滤掉太小的簇
        result = [
            cluster for cluster in clusters.values()
            if len(cluster) >= self.config.MIN_PHOTOS_PER_EVENT
        ]

        return result

    def create_events_from_clusters(
        self,
        clusters: List[List[int]],
        photo_data_list: List[PhotoData],
        user_id: int,
        db: Session
    ) -> List[Event]:
        """
        从聚类结果创建事件
        """
        events = []

        for cluster_indices in clusters:
            # 获取簇中的照片数据
            cluster_photos = [photo_data_list[i] for i in cluster_indices]

            # 计算事件的起止时间
            timestamps = [p.shoot_time for p in cluster_photos if p.shoot_time]
            if not timestamps:
                continue

            start_time = min(timestamps)
            end_time = max(timestamps)

            # 计算中心坐标（使用有 GPS 的照片）
            gps_photos = [p for p in cluster_photos if p.has_gps]
            if gps_photos:
                center_lat, center_lon = calculate_center_point(
                    [(p.lat, p.lon) for p in gps_photos]
                )
            else:
                center_lat = center_lon = None

            # 选择封面照片（中间那张）
            cover_photo_id = cluster_photos[len(cluster_photos) // 2].id

            # 创建事件
            event = Event(
                user_id=user_id,
                title=f"旅行事件",  # 后续由 AI 生成
                location_name=None,  # 后续由地理编码生成
                gps_lat=center_lat,
                gps_lon=center_lon,
                start_time=start_time,
                end_time=end_time,
                photo_count=len(cluster_photos),
                cover_photo_id=cover_photo_id,
                status="clustered"
            )

            db.add(event)
            db.flush()  # 获取 event.id

            # 更新照片的 event_id
            for photo in cluster_photos:
                db.query(Photo).filter(Photo.id == photo.id).update(
                    {"event_id": event.id}
                )

            events.append(event)

        db.commit()

        return events


def cluster_user_photos(
    user_id: int,
    db: Session,
    config: ClusteringConfig = None
) -> List[Event]:
    """
    聚类用户的所有照片

    Args:
        user_id: 用户 ID
        db: 数据库会话
        config: 聚类配置

    Returns:
        生成的事件列表
    """
    # 1. 获取所有未聚类的照片
    photos = db.query(Photo).filter(
        Photo.user_id == user_id,
        Photo.event_id.is_(None),
        Photo.status == "uploaded"
    ).all()

    if not photos:
        return []

    # 2. 执行聚类
    clustering = SpacetimeClustering(config)
    photo_data_list = clustering.prepare_photo_data(photos)
    clusters = clustering.cluster(photo_data_list)

    # 3. 创建事件
    events = clustering.create_events_from_clusters(
        clusters, photo_data_list, user_id, db
    )

    return events


def recluster_event(
    event_id: int,
    db: Session
) -> Optional[Event]:
    """
    重新聚类某个事件（例如添加了新照片后）

    Args:
        event_id: 事件 ID
        db: 数据库会话

    Returns:
        新的事件，如果聚类失败返回 None
    """
    # 获取事件和关联的照片
    event = db.query(Event).filter(Event.id == event_id).first()
    if not event:
        return None

    photos = db.query(Photo).filter(
        Photo.event_id == event_id
    ).all()

    if len(photos) < 5:  # 照片太少，无法形成事件
        # 删除事件
        db.delete(event)
        db.commit()
        return None

    # 释放旧事件关联
    for photo in photos:
        photo.event_id = None

    # 重新聚类
    clustering = SpacetimeClustering()
    photo_data_list = clustering.prepare_photo_data(photos)
    clusters = clustering.cluster(photo_data_list)

    if not clusters:
        # 聚类失败，恢复关联
        for photo in photos:
            photo.event_id = event_id
        db.commit()
        return event

    # 创建新事件
    new_events = clustering.create_events_from_clusters(
        clusters, photo_data_list, event.user_id, db
    )

    # 删除旧事件
    db.delete(event)
    db.commit()

    return new_events[0] if new_events else None
```
