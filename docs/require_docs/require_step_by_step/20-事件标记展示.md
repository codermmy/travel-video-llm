### 2.1 事件标记组件 `src/components/map/EventMarker.tsx`

```typescript
/**
 * 事件标记组件
 */
import React, { memo } from 'react';
import { View, Image, Text, StyleSheet } from 'react-native';
import type { Event } from '@/types';

interface EventMarkerProps {
  event: Event;
  isFocused: boolean;
  onPress: () => void;
}

const EventMarker = ({ event, isFocused, onPress }: EventMarkerProps) => {
  return (
    <View style={[styles.container, isFocused && styles.focused]}>
      <Image
        source={{ uri: event.coverPhotoUrl || require('@/assets/placeholder.png') }}
        style={styles.image}
      />
      <View style={styles.label}>
        <Text style={styles.title}>{event.title}</Text>
        <Text style={styles.date}>{formatDate(event.startTime)}</Text>
      </View>
    </View>
  );
};

export default memo(EventMarker);
```

### 2.2 事件气泡弹窗

```typescript
/**
 * 事件气泡组件
 */
import React from 'react';
import { View, Text, StyleSheet, Pressable } from 'react-native';

interface EventBubbleProps {
  event: Event;
  onClose: () => void;
  onViewDetails: () => void;
}

export function EventBubble({ event, onClose, onViewDetails }: EventBubbleProps) {
  return (
    <View style={styles.bubble}>
      <Image source={{ uri: event.coverPhotoUrl }} style={styles.cover} />
      <Text style={styles.title}>{event.title}</Text>
      <Text style={styles.date}>{formatDateRange(event.startTime, event.endTime)}</Text>
      <Text style={styles.count}>{event.photoCount} 张照片</Text>
      <Pressable style={styles.button} onPress={onViewDetails}>
        <Text style={styles.buttonText}>查看详情</Text>
      </Pressable>
      <Pressable style={styles.closeButton} onPress={onClose}>
        <Text style={styles.closeText}>✕</Text>
      </Pressable>
    </View>
  );
}
```
