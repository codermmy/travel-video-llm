#### `app/integrations/tongyi.py`

```python
"""
通义千问/万相 API 集成
"""
import os
import base64
import requests
from typing import Optional, List
from PIL import Image
from io import BytesIO

from app.config import get_settings

settings = get_settings()


class TongyiClient:
    """通义千问/万相 API 客户端"""

    def __init__(self):
        self.api_key = settings.DASHSCOPE_API_KEY
        self.base_url = "https://dashscope.aliyuncs.com/api/v1/services/aigc"
        self.timeout = 30

    def _get_headers(self) -> dict:
        """获取请求头"""
        return {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }

    def is_configured(self) -> bool:
        """检查是否已配置 API Key"""
        return bool(self.api_key)

    def analyze_image(
        self,
        image_url: str,
        prompt: str = "请描述这张照片的内容"
    ) -> Optional[dict]:
        """
        图像内容分析（通义万相）

        Args:
            image_url: 图片 URL
            prompt: 提示词

        Returns:
            分析结果，包含描述、标签、情感
        """
        if not self.is_configured():
            print("警告：通义 API Key 未配置")
            return None

        url = f"{self.base_url}/multimodal-generation/generation"

        payload = {
            "model": "qwen-vl-max",
            "input": {
                "messages": [
                    {
                        "role": "system",
                        "content": "你是一个专业的图像分析助手。"
                    },
                    {
                        "role": "user",
                        "content": [
                            {"image": image_url},
                            {"text": prompt}
                        ]
                    }
                ]
            }
        }

        try:
            response = requests.post(
                url,
                json=payload,
                headers=self._get_headers(),
                timeout=self.timeout
            )
            response.raise_for_status()

            data = response.json()

            if data.get("output"):
                # 解析结果
                content = data["output"]["text"]
                return self._parse_image_result(content)

            return None

        except requests.RequestException as e:
            print(f"通义万相 API 请求失败: {e}")
            return None
        except Exception as e:
            print(f"图像分析失败: {e}")
            return None

    def _parse_image_result(self, content: str) -> dict:
        """
        解析图像分析结果

        尝试从返回内容中提取结构化信息
        """
        # 简化版：直接返回原始内容
        # 后续可以优化为使用 JSON 返回格式
        return {
            "description": content,
            "tags": [],
            "emotion": self._detect_emotion(content)
        }

    def _detect_emotion(self, text: str) -> str:
        """
        从文本中检测情感标签
        """
        text_lower = text.lower()

        # 关键词匹配
        emotion_keywords = {
            "Happy": ["开心", "快乐", "笑容", "欢笑", "欢乐", "愉快", "幸福"],
            "Calm": ["宁静", "安静", "平静", "悠闲", "舒适", "放松", "平和"],
            "Epic": ["壮观", "宏大", "雄伟", "震撼", "气势", "宏伟", "辽阔"],
            "Romantic": ["浪漫", "温馨", "甜蜜", "温柔", "夕阳", "花朵", "美丽"]
        }

        scores = {}
        for emotion, keywords in emotion_keywords.items():
            score = sum(1 for kw in keywords if kw in text_lower)
            if score > 0:
                scores[emotion] = score

        if not scores:
            return "Calm"  # 默认

        return max(scores, key=scores.get)

    def generate_story(
        self,
        prompt: str,
        max_tokens: int = 500
    ) -> Optional[str]:
        """
        文本生成（通义千问）

        Args:
            prompt: 提示词
            max_tokens: 最大 token 数

        Returns:
            生成的文本
        """
        if not self.is_configured():
            print("警告：通义 API Key 未配置")
            return None

        url = f"{self.base_url}/text-generation/generation"

        payload = {
            "model": "qwen-turbo",
            "input": {
                "messages": [
                    {
                        "role": "system",
                        "content": "你是一个旅行故事创作助手。"
                    },
                    {
                        "role": "user",
                        "content": prompt
                    }
                ]
            },
            "parameters": {
                "max_tokens": max_tokens,
                "temperature": 0.7
            }
        }

        try:
            response = requests.post(
                url,
                json=payload,
                headers=self._get_headers(),
                timeout=self.timeout
            )
            response.raise_for_status()

            data = response.json()

            if data.get("output"):
                return data["output"]["text"]

            return None

        except requests.RequestException as e:
            print(f"通义千问 API 请求失败: {e}")
            return None
        except Exception as e:
            print(f"文本生成失败: {e}")
            return None

    def generate_event_story(
        self,
        location: str,
        date_range: str,
        photo_descriptions: List[str]
    ) -> Optional[dict]:
        """
        生成事件故事

        Args:
            location: 地点
            date_range: 时间范围
            photo_descriptions: 照片描述列表

        Returns:
            故事生成结果
        """
        # 构建提示词
        desc_text = "\n".join([f"- {d}" for d in photo_descriptions])

        prompt = f"""请根据以下信息生成一个简短的旅行故事：

地点：{location}
时间：{date_range}
照片描述：
{desc_text}

要求：
1. 100-200字
2. 情感基调：宁静、放松
3. 包含地点信息
4. 语言优美，有故事感

请以 JSON 格式返回：
{{
  "title": "事件标题",
  "story": "故事内容",
  "emotion": "情感标签（Happy/Calm/Epic/Romantic）"
}}
"""

        response_text = self.generate_story(prompt, max_tokens=800)

        if not response_text:
            return None

        # 尝试解析 JSON
        try:
            import json
            # 提取 JSON 部分
            if "```json" in response_text:
                json_start = response_text.index("```json") + 7
                json_end = response_text.index("```", json_start)
                json_str = response_text[json_start:json_end]
            elif "{" in response_text:
                json_start = response_text.index("{")
                json_end = response_text.rindex("}") + 1
                json_str = response_text[json_start:json_end]
            else:
                return None

            result = json.loads(json_str)
            return result

        except:
            pass

        # 解析失败，返回默认格式
        return {
            "title": f"{location}之旅",
            "story": response_text[:200] if len(response_text) > 200 else response_text,
            "emotion": "Calm"
        }

#### `app/services/ai_service.py`

```python
"""
AI 服务
"""
from typing import Optional, List
from app.integrations.tongyi import tongyi_client
from app.utils.cache import geocoding_cache


class AIService:
    """AI 服务"""

    def __init__(self):
        self.client = tongyi_client
        self.cache_ttl = 86400  # 缓存 24 小时

    def generate_event_story(
        self,
        event_id: int,
        location: str,
        start_time: str,
        end_time: str,
        photo_descriptions: List[str]
    ) -> Optional[dict]:
        """
        为事件生成故事

        Args:
            event_id: 事件 ID
            location: 地点
            start_time: 开始时间
            end_time: 结束时间
            photo_descriptions: 照片描述

        Returns:
            生成结果 {title, story, emotion}
        """
        # 检查缓存
        cache_key = f"story:{event_id}"
        cached = geocoding_cache.get(cache_key)
        if cached:
            return cached

        # 格式化时间范围
        from datetime import datetime
        start = datetime.fromisoformat(start_time)
        end = datetime.fromisoformat(end_time)
        date_range = f"{start.strftime('%m月%d日')} - {end.strftime('%m月%d日')}"

        # 调用 API
        result = self.client.generate_event_story(
            location=location,
            date_range=date_range,
            photo_descriptions=photo_descriptions
        )

        if result:
            # 缓存结果
            geocoding_cache.set(cache_key, result, self.cache_ttl)

        return result

    def analyze_photo_batch(
        self,
        photo_urls: List[str],
        prompt: str = "请描述这张照片的内容"
    ) -> List[Optional[dict]]:
        """
        批量分析照片

        Args:
            photo_urls: 照片 URL 列表
            prompt: 提示词

        Returns:
            分析结果列表
        """
        results = []

        for url in photo_urls:
            result = self.client.analyze_image(url, prompt)
            results.append(result)

        return results


# 导出服务实例
ai_service = AIService()
```
