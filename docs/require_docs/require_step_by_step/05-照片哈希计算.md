### 3.4 完整代码实现

#### `src/utils/hashUtils.ts`

```typescript
/**
 * 哈希计算工具函数
 */
import * as Crypto from 'expo-crypto';
import * as FileSystem from 'expo-file-system';

/**
 * 计算文件的 SHA-256 哈希值
 *
 * @param fileUri - 文件 URI（支持 content:// 和 file://）
 * @returns SHA-256 哈希值（64位十六进制字符串）
 * @throws 文件读取失败或哈希计算失败
 *
 * @example
 * const hash = await calculateFileHash('content://media/external/images/123');
 * console.log(hash); // "a3d5f8e9b2c1..."
 */
export async function calculateFileHash(fileUri: string): Promise<string> {
  try {
    // 1. 读取文件内容
    // 对于大文件，可以考虑分块读取
    const fileInfo = await FileSystem.getInfoAsync(fileUri);

    if (!fileInfo.exists) {
      throw new Error(`文件不存在: ${fileUri}`);
    }

    // expo-crypto 的 digestAsync 直接支持文件 URI
    const digest = await Crypto.digestStringAsync(
      Crypto.CryptoDigestAlgorithm.SHA256,
      fileUri,
      { encoding: 'hex' }
    );

    return digest;
  } catch (error) {
    console.error('计算哈希失败:', fileUri, error);
    throw new Error(`计算哈希失败: ${error instanceof Error ? error.message : String(error)}`);
  }
}

/**
 * 批量计算多个文件的哈希值
 *
 * @param fileUris - 文件 URI 数组
 * @param onProgress - 进度回调 (current, total)
 * @returns Promise<{ uri: string; hash: string }[]>
 *
 * @example
 * const results = await calculateFileHashes(
 *   ['uri1', 'uri2', 'uri3'],
 *   (current, total) => console.log(`${current}/${total}`)
 * );
 */
export async function calculateFileHashes(
  fileUris: string[],
  onProgress?: (current: number, total: number) => void
): Promise<Array<{ uri: string; hash: string }>> {
  const results: Array<{ uri: string; hash: string }> = [];
  const total = fileUris.length;

  for (let i = 0; i < total; i++) {
    const uri = fileUris[i];

    try {
      const hash = await calculateFileHash(uri);
      results.push({ uri, hash });

      // 报告进度
      if (onProgress) {
        onProgress(i + 1, total);
      }
    } catch (error) {
      console.error(`计算哈希失败: ${uri}`, error);
      // 失败的文件也加入结果，hash 为空字符串
      results.push({ uri, hash: '' });

      if (onProgress) {
        onProgress(i + 1, total);
      }
    }
  }

  return results;
}

/**
 * 比较两个哈希值是否相等
 *
 * @param hash1 - 第一个哈希值
 * @param hash2 - 第二个哈希值
 * @returns 是否相等
 */
export function compareHashes(hash1: string, hash2: string): boolean {
  // 统一转小写比较
  return hash1.toLowerCase() === hash2.toLowerCase();
}

/**
 * 验证哈希格式
 *
 * SHA-256 哈希应该是 64 位十六进制字符
 *
 * @param hash - 哈希值
 * @returns 是否有效
 */
export function isValidHash(hash: string): boolean {
  return /^[a-f0-9]{64}$/i.test(hash);
}
```

#### `src/services/album/photoHasher.ts`

```typescript
/**
 * 照片哈希服务 - 封装相册照片的哈希计算
 */
import * as ImagePicker from 'expo-image-picker';
import { calculateFileHash, calculateFileHashes } from '@/utils/hashUtils';

export interface PhotoHashResult {
  uri: string;
  hash: string;
  width: number;
  height: number;
}

/**
 * 计算单个照片的哈希值（带尺寸信息）
 *
 * @param asset - ImagePicker 资产对象
 * @returns 照片哈希结果
 */
export async function calculatePhotoHash(
  asset: ImagePicker.ImagePickerAsset
): Promise<PhotoHashResult> {
  const hash = await calculateFileHash(asset.uri);

  return {
    uri: asset.uri,
    hash,
    width: asset.width || 0,
    height: asset.height || 0,
  };
}

/**
 * 批量计算照片哈希值
 *
 * @param assets - ImagePicker 资产数组
 * @param onProgress - 进度回调
 * @returns 照片哈希结果数组
 */
export async function calculatePhotoHashes(
  assets: ImagePicker.ImagePickerAsset[],
  onProgress?: (current: number, total: number) => void
): Promise<PhotoHashResult[]> {
  const uris = assets.map((a) => a.uri);
  const hashResults = await calculateFileHashes(uris, onProgress);

  return hashResults.map((result) => {
    const asset = assets.find((a) => a.uri === result.uri);
    return {
      uri: result.uri,
      hash: result.hash,
      width: asset?.width || 0,
      height: asset?.height || 0,
    };
  });
}

/**
 * 过滤已有哈希（用于去重）
 *
 * @param newHashes - 新计算的哈希结果
 * @param existingHashes - 已存在的哈希集合
 * @returns 需要上传的新照片
 */
export function filterNewPhotos(
  newHashes: PhotoHashResult[],
  existingHashes: Set<string>
): PhotoHashResult[] {
  return newHashes.filter((item) => {
    if (!item.hash) {
      // 哈希计算失败的，保守起见也算新照片
      return true;
    }
    return !existingHashes.has(item.hash);
  });
}
```
