## 1. 任务目标

实现照片缩略图生成功能，用于减少上传流量和存储空间。

**核心功能**：
- 压缩照片到指定尺寸（宽度 1080px）
- 控制 JPEG 质量（80%）
- 目标大小约 200KB/张
- 保存到本地缓存目录
- 支持批量处理

#### `src/utils/imageUtils.ts`

```typescript
/**
 * 图片处理工具函数
 */
import * as ImageManipulator from 'expo-image-manipulator';
import * as FileSystem from 'expo-file-system';
import { THUMBNAIL_CONFIG } from '@/constants';

export interface ThumbnailResult {
  uri: string;          // 缩略图本地 URI
  width: number;        // 实际宽度
  height: number;       // 实际高度
  size: number;         // 文件大小（字节）
}

/**
 * 生成缩略图
 *
 * @param sourceUri - 原图 URI
 * @param options - 配置选项
 * @returns 缩略图结果
 *
 * @example
 * const thumbnail = await generateThumbnail('content://...');
 * console.log(thumbnail.size); // 约 200KB
 */
export async function generateThumbnail(
  sourceUri: string,
  options: {
    width?: number;
    quality?: number;
    maxSize?: number;
  } = {}
): Promise<ThumbnailResult> {
  const {
    width = THUMBNAIL_CONFIG.WIDTH,
    quality = THUMBNAIL_CONFIG.QUALITY,
    maxSize = THUMBNAIL_CONFIG.MAX_SIZE,
  } = options;

  try {
    // 1. 获取原图尺寸
    const fileInfo = await FileSystem.getInfoAsync(sourceUri);
    if (!fileInfo.exists) {
      throw new Error(`文件不存在: ${sourceUri}`);
    }

    // 2. 压缩图片
    const result = await ImageManipulator.manipulateAsync(
      sourceUri,
      [
        // 调整大小（宽度固定，高度自适应）
        {
          resize: {
            width,
          },
        },
      ],
      {
        compress: quality,
        format: ImageManipulator.SaveFormat.JPEG,
      }
    );

    // 3. 获取文件大小
    const thumbnailInfo = await FileSystem.getInfoAsync(result.uri);
    const size = thumbnailInfo.size || 0;

    // 4. 如果超过最大大小，继续降低质量
    if (size > maxSize) {
      return await compressToSize(sourceUri, width, maxSize);
    }

    return {
      uri: result.uri,
      width: result.width,
      height: result.height,
      size,
    };
  } catch (error) {
    console.error('生成缩略图失败:', sourceUri, error);
    throw new Error(`生成缩略图失败: ${error instanceof Error ? error.message : String(error)}`);
  }
}

/**
 * 压缩到指定大小以内
 *
 * @param sourceUri - 原图 URI
 * @param width - 目标宽度
 * @param maxSize - 最大文件大小
 * @param minQuality - 最低质量（默认 0.5）
 * @returns 缩略图结果
 */
async function compressToSize(
  sourceUri: string,
  width: number,
  maxSize: number,
  minQuality: number = 0.5
): Promise<ThumbnailResult> {
  let quality = THUMBNAIL_CONFIG.QUALITY;
  let result: ThumbnailResult | null = null;

  // 二分查找合适的压缩质量
  while (quality >= minQuality) {
    const manipResult = await ImageManipulator.manipulateAsync(
      sourceUri,
      [{ resize: { width } }],
      {
        compress: quality,
        format: ImageManipulator.SaveFormat.JPEG,
      }
    );

    const info = await FileSystem.getInfoAsync(manipResult.uri);
    const size = info.size || 0;

    result = {
      uri: manipResult.uri,
      width: manipResult.width,
      height: manipResult.height,
      size,
    };

    if (size <= maxSize) {
      break;
    }

    // 降低质量重试
    quality -= 0.1;
  }

  if (!result) {
    throw new Error('无法压缩到目标大小');
  }

  return result;
}

/**
 * 批量生成缩略图
 *
 * @param sourceUris - 原图 URI 数组
 * @param options - 配置选项
 * @param onProgress - 进度回调
 * @returns 缩略图结果数组
 */
export async function generateThumbnails(
  sourceUris: string[],
  options: {
    width?: number;
    quality?: number;
    maxSize?: number;
  } = {},
  onProgress?: (current: number, total: number) => void
): Promise<ThumbnailResult[]> {
  const results: ThumbnailResult[] = [];
  const total = sourceUris.length;
  const errors: Array<{ uri: string; error: string }> = [];

  for (let i = 0; i < total; i++) {
    const uri = sourceUris[i];

    try {
      const thumbnail = await generateThumbnail(uri, options);
      results.push(thumbnail);

      if (onProgress) {
        onProgress(i + 1, total);
      }
    } catch (error) {
      const errorMsg = error instanceof Error ? error.message : String(error);
      errors.push({ uri, error: errorMsg });

      if (onProgress) {
        onProgress(i + 1, total);
      }
    }
  }

  // 如果有错误，打印日志
  if (errors.length > 0) {
    console.warn('部分缩略图生成失败:', errors);
  }

  return results;
}

/**
 * 格式化文件大小显示
 *
 * @param bytes - 字节数
 * @returns 格式化的字符串，如 "1.5 MB"
 */
export function formatFileSize(bytes: number): string {
  if (bytes < FILE_SIZE_UNITS.KB) {
    return `${bytes} B`;
  }

  if (bytes < FILE_SIZE_UNITS.MB) {
    const kb = bytes / FILE_SIZE_UNITS.KB;
    return `${kb.toFixed(1)} KB`;
  }

  const mb = bytes / FILE_SIZE_UNITS.MB;
  return `${mb.toFixed(2)} MB`;
}

const FILE_SIZE_UNITS = {
  KB: 1024,
  MB: 1024 * 1024,
  GB: 1024 * 1024 * 1024,
};
```

#### `src/services/album/thumbnailGenerator.ts`

```typescript
/**
 * 缩略图生成服务
 */
import * as FileSystem from 'expo-file-system';
import { generateThumbnails, formatFileSize } from '@/utils/imageUtils';
import { THUMBNAIL_CONFIG } from '@/constants';

// 缩略图缓存目录
const THUMBNAIL_DIR = `${FileSystem.documentDirectory}thumbnails/`;

/**
 * 确保缩略图目录存在
 */
async function ensureThumbnailDir(): Promise<void> {
  const dirInfo = await FileSystem.getInfoAsync(THUMBNAIL_DIR);

  if (!dirInfo.exists) {
    await FileSystem.makeDirectoryAsync(THUMBNAIL_DIR, { intermediates: true });
  }
}

/**
 * 生成缩略图文件名
 *
 * @param hash - 照片哈希值
 * @returns 缩略图文件路径
 */
export function getThumbnailPath(hash: string): string {
  return `${THUMBNAIL_DIR}${hash}.jpg`;
}

/**
 * 检查缩略图是否已存在
 *
 * @param hash - 照片哈希值
 * @returns 是否存在
 */
export async function hasThumbnail(hash: string): Promise<boolean> {
  const thumbnailPath = getThumbnailPath(hash);
  const info = await FileSystem.getInfoAsync(thumbnailPath);
  return info.exists;
}

/**
 * 获取已存在的缩略图路径
 *
 * @param hash - 照片哈希值
 * @returns 缩略图 URI 或 null
 */
export async function getThumbnailUri(hash: string): Promise<string | null> {
  const exists = await hasThumbnail(hash);
  return exists ? getThumbnailPath(hash) : null;
}

/**
 * 生成并保存缩略图
 *
 * @param sourceUri - 原图 URI
 * @param hash - 照片哈希值
 * @returns 缩略图结果
 */
export async function generateAndSaveThumbnail(
  sourceUri: string,
  hash: string
): Promise<{
  uri: string;
  width: number;
  height: number;
  size: number;
  sizeFormatted: string;
}> {
  // 确保目录存在
  await ensureThumbnailDir();

  // 检查是否已存在
  const existing = await getThumbnailUri(hash);
  if (existing) {
    const info = await FileSystem.getInfoAsync(existing);
    return {
      uri: existing,
      width: 0,
      height: 0,
      size: info.size || 0,
      sizeFormatted: formatFileSize(info.size || 0),
    };
  }

  // 生成新的缩略图
  const thumbnail = await generateThumbnail(sourceUri, {
    width: THUMBNAIL_CONFIG.WIDTH,
    quality: THUMBNAIL_CONFIG.QUALITY,
    maxSize: THUMBNAIL_CONFIG.MAX_SIZE,
  });

  // 移动到缩略图目录
  const targetPath = getThumbnailPath(hash);

  // 如果源和目标不同，复制文件
  if (thumbnail.uri !== targetPath) {
    await FileSystem.copyAsync({
      from: thumbnail.uri,
      to: targetPath,
    });

    // 删除临时文件
    try {
      await FileSystem.deleteAsync(thumbnail.uri);
    } catch {
      // 忽略删除失败
    }
  }

  return {
    uri: targetPath,
    width: thumbnail.width,
    height: thumbnail.height,
    size: thumbnail.size,
    sizeFormatted: formatFileSize(thumbnail.size),
  };
}

/**
 * 批量生成缩略图
 *
 * @param photos - 照片数组 { uri, hash }
 * @param onProgress - 进度回调
 * @returns 缩略图结果数组
 */
export async function generateThumbnailsForPhotos(
  photos: Array<{ uri: string; hash: string }>,
  onProgress?: (current: number, total: number) => void
): Promise<Array<{
  hash: string;
  uri: string;
  size: number;
  sizeFormatted: string;
}>> {
  await ensureThumbnailDir();

  const results: Array<{
    hash: string;
    uri: string;
    size: number;
    sizeFormatted: string;
  }> = [];

  const total = photos.length;

  for (let i = 0; i < total; i++) {
    const { uri, hash } = photos[i];

    try {
      const thumbnail = await generateAndSaveThumbnail(uri, hash);
      results.push({
        hash,
        uri: thumbnail.uri,
        size: thumbnail.size,
        sizeFormatted: thumbnail.sizeFormatted,
      });

      if (onProgress) {
        onProgress(i + 1, total);
      }
    } catch (error) {
      console.error(`生成缩略图失败: ${hash}`, error);

      if (onProgress) {
        onProgress(i + 1, total);
      }
    }
  }

  return results;
}

/**
 * 清除缩略图缓存
 *
 * @returns 清除的文件数量
 */
export async function clearThumbnailCache(): Promise<number> {
  try {
    const files = await FileSystem.readDirectoryAsync(THUMBNAIL_DIR);
    let count = 0;

    for (const file of files) {
      const filePath = `${THUMBNAIL_DIR}${file}`;
      await FileSystem.deleteAsync(filePath);
      count++;
    }

    return count;
  } catch (error) {
    console.error('清除缩略图缓存失败:', error);
    return 0;
  }
}

/**
 * 获取缩略图缓存总大小
 *
 * @returns 缓存大小（字节）
 */
export async function getThumbnailCacheSize(): Promise<number> {
  try {
    const files = await FileSystem.readDirectoryAsync(THUMBNAIL_DIR);
    let totalSize = 0;

    for (const file of files) {
      const filePath = `${THUMBNAIL_DIR}${file}`;
      const info = await FileSystem.getInfoAsync(filePath);
      totalSize += info.size || 0;
    }

    return totalSize;
  } catch {
    return 0;
  }
}
```
