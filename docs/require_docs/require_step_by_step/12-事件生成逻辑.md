#### `app/models/event.py`

```python
"""
事件数据模型
"""
from sqlalchemy import Column, Integer, String, Text, DateTime, ForeignKey, DECIMAL
from sqlalchemy.orm import relationship
from datetime import datetime

from app.models.base import Base


class Event(Base):
    """事件表"""

    __tablename__ = "events"

    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, ForeignKey("users.id"), nullable=False)
    title = Column(String(200))  # AI 生成的标题
    location_name = Column(String(100))  # 逆向地理编码结果
    gps_lat = Column(DECIMAL(10, 7))  # 中心坐标
    gps_lon = Column(DECIMAL(10, 7))
    start_time = Column(DateTime)  # 开始时间
    end_time = Column(DateTime)  # 结束时间
    photo_count = Column(Integer, default=0)  # 照片数量
    cover_photo_id = Column(Integer, ForeignKey("photos.id"))
    cover_photo_url = Column(String(500))  # 封面 URL
    story_text = Column(Text)  # AI 生成的故事
    emotion_tag = Column(String(20))  # Happy/Calm/Epic/Romantic
    music_id = Column(String(100))  # 音乐 ID
    music_url = Column(String(500))  # 音乐 URL
    status = Column(String(20), default="clustered")  # clustered/generated
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 关系
    user = relationship("User", back_populates="events")
    photos = relationship("Photo", back_populates="event", foreign_keys=[Photo.event_id])

    def __repr__(self):
        return f"<Event(id={self.id}, title={self.title})>"
```

```python
    title: str
    locationName: Optional[str]
    gpsLat: Optional[float]
    gpsLon: Optional[float]

    photoCount: int
    coverPhotoUrl: Optional[str]

    emotionTag: Optional[str]

    status: str

    class Config:

    """事件列表响应"""
    items: List[EventResponse]
    total: int
    page: int
    pageSize: int
    totalPages: int

    """事件详情（包含照片列表）"""
```

### 3.4 事件服务

#### `app/services/event_service.py`

```python
"""
事件业务逻辑
"""
from typing import List, Optional
from sqlalchemy.orm import Session
from sqlalchemy import func

from app.models.event import Event
from app.models.photo import Photo
from app.schemas.event import EventUpdateRequest


class EventService:
    """事件服务"""

    @staticmethod
    def get_user_events(
        user_id: int,
        db: Session,
        page: int = 1,
        page_size: int = 20
    ) -> tuple[List[Event], int]:
        """
        获取用户的事件列表

        Returns:
            (事件列表, 总数)
        """
        query = db.query(Event).filter(Event.user_id == user_id)

        total = query.count()

        # 按开始时间倒序
        events = query.order_by(Event.start_time.desc())\
            .offset((page - 1) * page_size)\
            .limit(page_size)\
            .all()

        return events, total

    @staticmethod
    def get_event_detail(
        event_id: int,
        user_id: int,
        db: Session
    ) -> Optional[Event]:
        """获取事件详情"""
        return db.query(Event).filter(
            Event.id == event_id,
            Event.user_id == user_id
        ).first()

    @staticmethod
    def get_event_photos(
        event_id: int,
        user_id: int,
        db: Session
    ) -> List[Photo]:
        """获取事件下的照片"""
        return db.query(Photo).filter(
            Photo.event_id == event_id,
            Photo.user_id == user_id
        ).order_by(Photo.shoot_time.asc()).all()

    @staticmethod
    def update_event(
        event_id: int,
        user_id: int,
        update_data: EventUpdateRequest,
        db: Session
    ) -> Optional[Event]:
        """更新事件"""
        event = db.query(Event).filter(
            Event.id == event_id,
            Event.user_id == user_id
        ).first()

        if not event:
            return None

        # 更新字段
        if update_data.title is not None:
            event.title = update_data.title
        if update_data.storyText is not None:
            event.story_text = update_data.storyText
        if update_data.emotionTag is not None:
            event.emotion_tag = update_data.emotionTag
        if update_data.musicId is not None:
            event.music_id = update_data.musicId
        if update_data.musicUrl is not None:
            event.music_url = update_data.musicUrl

        db.commit()
        db.refresh(event)

        return event

    @staticmethod
    def delete_event(
        event_id: int,
        user_id: int,
        db: Session
    ) -> bool:
        """删除事件（保留照片，取消关联）"""
        event = db.query(Event).filter(
            Event.id == event_id,
            Event.user_id == user_id
        ).first()

        if not event:
            return False

        # 取消照片关联
        db.query(Photo).filter(Photo.event_id == event_id).update(
            {"event_id": None}
        )

        # 删除事件
        db.delete(event)
        db.commit()

        return True

    @staticmethod
    def get_event_stats(
        user_id: int,
        db: Session
    ) -> dict:
        """获取用户事件统计"""
        total = db.query(Event).filter(Event.user_id == user_id).count()

        # 按情感标签统计
        emotion_stats = db.query(
            Event.emotion_tag,
            func.count(Event.id)
        ).filter(
            Event.user_id == user_id,
            Event.emotion_tag.isnot(None)
        ).group_by(Event.emotion_tag).all()

        # 按状态统计
        clustered = db.query(Event).filter(
            Event.user_id == user_id,
            Event.status == "clustered"
        ).count()

        generated = db.query(Event).filter(
            Event.user_id == user_id,
            Event.status == "generated"
        ).count()

        return {
            "total": total,
            "byEmotion": {e[0]: e[1] for e in emotion_stats},
            "clustered": clustered,
            "generated": generated,
        }


# 导出服务实例
event_service = EventService()
```

### 3.5 事件接口

#### `app/api/v1/events.py`

```python
"""
事件接口
"""
from fastapi import APIRouter, Depends, HTTPException, status, Query
from sqlalchemy.orm import Session

from app.models.base import get_db
from app.schemas.event import (
    EventResponse, EventListResponse,
    EventDetailResponse, EventUpdateRequest
)
from app.schemas.common import ApiResponse
from app.api.deps import CurrentUserIdDep
from app.services.event_service import event_service
from datetime import datetime

router = APIRouter()


@router.get("", response_model=ApiResponse[EventListResponse])
async def get_events(
    user_id: CurrentUserIdDep,
    page: int = Query(1, ge=1),
    pageSize: int = Query(20, ge=1, le=100),
    db: Session = Depends(get_db)
):
    """获取用户事件列表"""
    events, total = event_service.get_user_events(user_id, db, page, pageSize)

    totalPages = (total + pageSize - 1) // pageSize

    return ApiResponse(
        success=True,
        data=EventListResponse(
            items=[
                EventResponse(
                    id=e.id,
                    title=e.title,
                    locationName=e.location_name,
                    gpsLat=float(e.gps_lat) if e.gps_lat else None,
                    gpsLon=float(e.gps_lon) if e.gps_lon else None,
                    startTime=e.start_time,
                    endTime=e.end_time,
                    photoCount=e.photo_count,
                    coverPhotoUrl=e.cover_photo_url,
                    storyText=e.story_text,
                    emotionTag=e.emotion_tag,
                    musicUrl=e.music_url,
                    status=e.status
                )
                for e in events
            ],
            total=total,
            page=page,
            pageSize=pageSize,
            totalPages=totalPages
        ),
        timestamp=datetime.utcnow()
    )


@router.get("/stats", response_model=ApiResponse[dict])
async def get_event_stats(
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """获取事件统计信息"""
    stats = event_service.get_event_stats(user_id, db)

    return ApiResponse(
        success=True,
        data=stats,
        timestamp=datetime.utcnow()
    )


@router.get("/{event_id}", response_model=ApiResponse[EventDetailResponse])
async def get_event_detail(
    event_id: int,
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """获取事件详情（包含照片列表）"""
    event = event_service.get_event_detail(event_id, user_id, db)

    if not event:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="事件不存在"
        )

    # 获取事件照片
    photos = event_service.get_event_photos(event_id, user_id, db)

    return ApiResponse(
        success=True,
        data=EventDetailResponse(
            id=event.id,
            title=event.title,
            locationName=event.location_name,
            gpsLat=float(event.gps_lat) if event.gps_lat else None,
            gpsLon=float(event.gps_lon) if event.gps_lon else None,
            startTime=event.start_time,
            endTime=event.end_time,
            photoCount=event.photo_count,
            coverPhotoUrl=event.cover_photo_url,
            storyText=event.story_text,
            emotionTag=event.emotion_tag,
            musicUrl=event.music_url,
            status=event.status,
            photos=[
                {
                    "id": p.id,
                    "thumbnailUrl": p.thumbnail_url,
                    "shootTime": p.shoot_time.isoformat() if p.shoot_time else None,
                }
                for p in photos
            ]
        ),
        timestamp=datetime.utcnow()
    )


@router.patch("/{event_id}", response_model=ApiResponse[EventResponse])
async def update_event(
    event_id: int,
    update_data: EventUpdateRequest,
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """更新事件信息"""
    event = event_service.update_event(event_id, user_id, update_data, db)

    if not event:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="事件不存在"
        )

    return ApiResponse(
        success=True,
        data=EventResponse(
            id=event.id,
            title=event.title,
            locationName=event.location_name,
            gpsLat=float(event.gps_lat) if event.gps_lat else None,
            gpsLon=float(event.gps_lon) if event.gps_lon else None,
            startTime=event.start_time,
            endTime=event.end_time,
            photoCount=event.photo_count,
            coverPhotoUrl=event.cover_photo_url,
            storyText=event.story_text,
            emotionTag=event.emotion_tag,
            musicUrl=event.music_url,
            status=event.status
        ),
        timestamp=datetime.utcnow()
    )


@router.delete("/{event_id}", response_model=ApiResponse[dict])
async def delete_event(
    event_id: int,
    user_id: CurrentUserIdDep,
    db: Session = Depends(get_db)
):
    """删除事件"""
    success = event_service.delete_event(event_id, user_id, db)

    if not success:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="事件不存在"
        )

    return ApiResponse(
        success=True,
        data={"message": "事件已删除"},
        timestamp=datetime.utcnow()
    )
```
