### 2.2 事件详情页 `app/(tabs)/events/[eventId]/index.tsx`

```typescript
/**
 * 事件详情页
 */
import { useLocalSearchParams, useRouter } from 'expo-router';
import { useQuery } from '@tanstack/react-query';
import { eventApi } from '@/services/api';
import { EventDetailScreen } from '@/screens';

export default function EventDetailPage() {
  const { eventId } = useLocalSearchParams();
  const router = useRouter();

  const { data: event, isLoading } = useQuery({
    queryKey: ['event', eventId],
    queryFn: () => eventApi.getEventDetail(Number(eventId)),
    enabled: !!eventId,
  });

  if (isLoading) {
    return <LoadingScreen />;
  }

  if (!event) {
    return <NotFoundScreen />;
  }

  return <EventDetailScreen event={event} />;
}
```

### 2.3 页面组件 `src/screens/EventDetailScreen/index.tsx`

```typescript
/**
 * 事件详情页面组件
 */
import React, { useState } from 'react';
import { View, Text, StyleSheet, ScrollView, Pressable, Image } from 'react-native';
import type { Event } from '@/types';

interface Props {
  event: Event;
}

export function EventDetailScreen({ event }: Props) {
  const [showSlideshow, setShowSlideshow] = useState(false);

  return (
    <ScrollView style={styles.container}>
      {/* 封面 */}
      <Image
        source={{ uri: event.coverPhotoUrl || require('@/assets/placeholder.png') }}
        style={styles.cover}
      />

      {/* 标题区域 */}
      <View style={styles.header}>
        <Text style={styles.title}>{event.title}</Text>
        <Text style={styles.location}>📍 {event.locationName || '未知地点'}</Text>
        <Text style={styles.date}>
          📅 {formatDateRange(event.startTime, event.endTime)}
        </Text>
      </View>

      {/* 故事 */}
      {event.storyText && (
        <View style={styles.storySection}>
          <Text style={styles.storyText}>{event.storyText}</Text>
        </View>
      )}

      {/* 音乐 */}
      {event.musicUrl && (
        <View style={styles.musicSection}>
          <Text style={styles.musicLabel}>🎵 背景音乐</Text>
          <Pressable style={styles.changeMusicBtn}>
            <Text style={styles.changeMusicText}>换一首</Text>
          </Pressable>
        </View>
      )}

      {/* 照片网格 */}
      <View style={styles.photosSection}>
        <Text style={styles.photosHeader}>照片 ({event.photoCount})</Text>
        <PhotoGrid photos={[]} />
      </View>

      {/* 幻灯片按钮 */}
      <Pressable style={styles.slideshowButton} onPress={() => setShowSlideshow(true)}>
        <Text style={styles.slideshowButtonText}>▶ 播放幻灯片</Text>
      </Pressable>
    </ScrollView>
  );
}
```
