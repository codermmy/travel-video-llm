### 2.1 进度组件

/**
 * 上传进度组件
 */
import React, { useEffect } from 'react';
import { View, Text, StyleSheet, ActivityIndicator, Pressable } from 'react-native';

interface UploadProgressProps {
  taskId: string;
  onComplete?: () => void;
}

export function UploadProgress({ taskId, onComplete }: UploadProgressProps) {
  const [progress, setProgress] = useState(0);
  const [status, setStatus] = useState<'uploading' | 'processing' | 'completed'>('uploading');

  useEffect(() => {
    // 轮询任务状态
    const interval = setInterval(async () => {
      const result = await fetchTaskStatus(taskId);
      if (result) {
        setProgress(result.progress || 0);
        setStatus(result.status || 'uploading');

        if (result.status === 'completed') {
          clearInterval(interval);
          onComplete?.();
        }
      }
    }, 2000);

    return () => clearInterval(interval);
  }, [taskId]);

  if (status === 'completed') {
    return null; // 或显示"完成"提示后消失
  }

  return (
    <View style={styles.overlay}>
      <View style={styles.modal}>
        <ActivityIndicator size="large" color="#6200EE" />
        <Text style={styles.status}>
          {status === 'uploading' ? '正在上传照片...' : '正在生成事件...'}
        </Text>
        <Text style={styles.progress}>{progress}%</Text>

        {status === 'processing' && (
          <Text style={styles.tip}>
            这可能需要几分钟时间
          </Text>
        )}

        <Pressable
          style={styles.backgroundButton}
          onPress={() => {/* 后台继续 */}}
        >
          <Text style={styles.backgroundButtonText}>后台继续</Text>
        </Pressable>
      </View>
    </View>
  );
}
