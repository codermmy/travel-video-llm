> **版本**: v2.0

> **状态**: 待开发

---

## 1. 概述

### 1.1 功能目标

实现邮箱 + 密码的登录/注册流程，作为用户账号体系的核心认证方式。此功能是用户开启"数字旅程"的第一站，需要在安全性与用户体验之间取得平衡。

### 1.2 设计理念

**概念**: "Journey Begin" —— 旅程的起点

- **情感基调**: 温暖、探索、期待感
- **视觉语言**: 柔和渐变 + 流动形态 + 地图等高线纹理
- **用户感受**: 安全、简洁、愉悦

### 1.3 功能范围

#### 必须实现
| 序号 | 功能 | 描述 |
|-----|------|------|
| 1 | 邮箱 + 密码登录 | 标准登录方式 |
| 2 | 邮箱 + 密码注册 | 新用户注册 |
| 3 | 登录/注册表单校验 | 邮箱格式、密码强度、密码一致性 |
| 4 | 密码可见性切换 | 眼睛图标显示/隐藏密码 |
| 5 | 错误状态展示 | 表单级、字段级错误提示 |
| 6 | 用户身份管理 | 登录后 user_id 作为数据归属主键 |
| 7 | OAuth 登录入口 | Google、Apple 登录（可选） |

#### 暂不实现
- 邮箱验证码验证
- 短信验证
- 找回密码功能（预留入口）

#### 密码校验

```typescript
// 密码规则
const PASSWORD_RULES = {
  minLength: 8,           // 最少 8 位
  requireLetter: true,    // 必须包含字母
  requireNumber: true,    // 必须包含数字
  noSpaces: true,         // 不允许空格
};

// 密码强度计算
function calculatePasswordStrength(password: string): {
  level: 'weak' | 'medium' | 'strong' | 'very-strong',
  score: number
} {
  let score = 0;
  if (password.length >= 8) score += 1;
  if (password.length >= 12) score += 1;
  if (/[a-z]/.test(password)) score += 1;
  if (/[A-Z]/.test(password)) score += 1;
  if (/[0-9]/.test(password)) score += 1;
  if (/[^a-zA-Z0-9]/.test(password)) score += 1;

  const levelMap: Record<number, string> = {
    0: 'weak',
    1: 'weak',
    2: 'medium',
    3: 'strong',
    4: 'strong',
    5: 'very-strong',
    6: 'very-strong',
  };

  return { level: levelMap[score] as any, score };
}

// 错误提示
- 空值: "请输入密码"
- 过短: "密码至少需要 8 位"
- 缺少字母/数字: "密码需包含字母和数字"
- 包含空格: "密码不能包含空格"
```

#### 确认密码校验

```typescript
// 校验逻辑
function validateConfirmPassword(password: string, confirmPassword: string): string | null {
  if (!confirmPassword) {
    return "请确认密码";
  }
  if (password !== confirmPassword) {
    return "两次输入的密码不一致";
  }
  return null;
}

// 校验时机
1. 实时（当两个字段都有值时）
2. 失去焦点
3. 提交表单时
```

### 4.2 错误处理规范

#### 错误提示层级

```
┌─────────────────────────────────────────────────────────┐
│                    错误提示层级                          │
├─────────────────────────────────────────────────────────┤
│  L1 - 字段级错误（输入框下方）                           │
│      • 实时校验错误                                      │
│      • 格式错误                                         │
│                                                         │
│  L2 - 表单级错误（页面顶部 Toast）                       │
│      • API 错误                                         │
│      • 网络错误                                         │
│                                                         │
│  L3 - 页面级错误（全屏提示）                             │
│      • 严重错误                                         │
│      • 需要用户操作                                     │
└─────────────────────────────────────────────────────────┘
```

#### 错误提示规范

| 场景 | 提示方式 | 持续时间 | 操作 |
|-----|---------|---------|------|
| 邮箱格式错误 | 字段下方提示 | 持续显示 | 输入正确后消失 |
| 密码格式错误 | 字段下方提示 | 持续显示 | 输入正确后消失 |

| 账号不存在 | 字段下方提示 + 快捷按钮 | 持续显示 | 点击跳转注册 |
| 密码错误 | 字段下方提示 | 持续显示 | 重新输入 |
| 邮箱已注册 | 页面顶部 Toast | 3秒 | 可手动关闭 |
| 网络错误 | 页面顶部 Toast | 3秒 | 可重试 |

// mobile/src/stores/authStore.ts

import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';

interface User {
  id: string;
  email: string;
  nickname?: string;
  avatar?: string;
}

interface AuthState {
  // 状态
  user: User | null;
  token: string | null;
  isAuthenticated: boolean;

  // 操作
  login: (email: string, password: string) => Promise<void>;
  register: (email: string, password: string) => Promise<void>;
  logout: () => void;
  updateUser: (user: Partial<User>) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      token: null,
      isAuthenticated: false,

      login: async (email, password) => {
        // 调用 API
        const response = await authApi.loginWithEmail({ email, password });

        set({
          user: response.data.user,
          token: response.data.token,
          isAuthenticated: true,
        });
      },

      register: async (email, password) => {
        const response = await authApi.registerWithEmail({ email, password });

        set({
          user: response.data.user,
          token: response.data.token,
          isAuthenticated: true,
        });
      },

      logout: () => {
        set({
          user: null,
          token: null,
          isAuthenticated: false,
        });
      },

      updateUser: (userData) => {
        const { user } = get();
        if (user) {
          set({ user: { ...user, ...userData } });
        }
      },
    }),
    {
      name: 'auth-storage',
      storage: createJSONStorage(() => AsyncStorage),
    }
  )
);

# backend/app/schemas/auth.py

from pydantic import BaseModel, EmailStr, Field, field_validator
from typing import Optional
from datetime import datetime

class UserBase(BaseModel):
    """用户基础信息"""
    email: EmailStr
    nickname: Optional[str] = None
    avatar: Optional[str] = None


class UserCreate(BaseModel):
    """用户创建请求"""
    email: EmailStr
    password: str = Field(..., min_length=8, max_length=100)

    @field_validator('password')
    @classmethod
    def validate_password(cls, v: str) -> str:
        """密码格式校验"""
        if not any(c.isalpha() for c in v):
            raise ValueError('密码必须包含字母')
        if not any(c.isdigit() for c in v):
            raise ValueError('密码必须包含数字')
        if ' ' in v:
            raise ValueError('密码不能包含空格')
        return v


class UserResponse(BaseModel):
    """用户响应"""
    id: str
    email: str
    nickname: Optional[str] = None
    avatar: Optional[str] = None
    created_at: datetime

    class Config:
        from_attributes = True


class LoginRequest(BaseModel):
    """登录请求"""
    email: EmailStr
    password: str = Field(..., min_length=1)


class AuthResponse(BaseModel):
    """认证响应"""
    user: UserResponse
    token: str
    refresh_token: Optional[str] = None


class ErrorResponse(BaseModel):
    """错误响应"""
    code: str
    message: str
    details: Optional[dict] = None

### 6.4 服务层实现

```python
# backend/app/services/auth_service.py

from passlib.context import CryptContext
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from datetime import datetime
import uuid

from app.models.user import User
from app.core.security import create_access_token, create_refresh_token
from app.schemas.auth import UserCreate, LoginRequest
from app.core.exceptions import ValidationError, NotFoundError, ConflictError

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


class AuthService:
    """认证服务"""

    def __init__(self, db: AsyncSession):
        self.db = db

    async def register(self, data: UserCreate) -> User:
        """
        用户注册

        Args:
            data: 注册请求数据

        Returns:
            User: 创建的用户对象

        Raises:
            ConflictError: 邮箱已存在
        """
        # 检查邮箱是否已存在
        existing = await self.db.execute(
            select(User).where(User.email == data.email)
        )
        if existing.scalar_one_or_none():
            raise ConflictError("该邮箱已被注册")

        # 创建用户
        user = User(
            id=str(uuid.uuid4()),
            email=data.email,
            hashed_password=self._hash_password(data.password),
            created_at=datetime.utcnow(),
        )

        self.db.add(user)
        await self.db.commit()
        await self.db.refresh(user)

        return user

    async def login(self, data: LoginRequest) -> User:
        """
        用户登录

        Args:
            data: 登录请求数据

        Returns:
            User: 登录成功的用户对象

        Raises:
            NotFoundError: 账号不存在
            ValidationError: 密码错误
        """
        # 查找用户
        result = await self.db.execute(
            select(User).where(User.email == data.email)
        )
        user = result.scalar_one_or_none()

        if not user:
            raise NotFoundError("账号不存在，请先注册")

        # 验证密码
        if not self._verify_password(data.password, user.hashed_password):
            raise ValidationError("密码错误，请重试")

        return user

    def create_tokens(self, user: User) -> dict:
        """创建访问令牌和刷新令牌"""
        access_token = create_access_token({"sub": user.id})
        refresh_token = create_refresh_token({"sub": user.id})
        return {
            "token": access_token,
            "refresh_token": refresh_token,
            "token_type": "bearer",
        }

    def _hash_password(self, password: str) -> str:
        """密码加密"""
        return pwd_context.hash(password)

    def _verify_password(self, plain_password: str, hashed_password: str) -> bool:
        """密码验证"""
        return pwd_context.verify(plain_password, hashed_password)
```

from fastapi import APIRouter, Depends, HTTPException, status, Request
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.deps import get_db
from app.schemas.auth import (
    LoginRequest,
    UserCreate,
    AuthResponse,
    ErrorResponse,
)
from app.services.auth_service import AuthService
from app.core.exceptions import AppException

router = APIRouter()


@router.post(
    "/login",
    response_model=AuthResponse,
    responses={
        404: {"model": ErrorResponse, "description": "账号不存在"},
        401: {"model": ErrorResponse, "description": "密码错误"},
        400: {"model": ErrorResponse, "description": "参数错误"},
    },
)
async def login(
    request: Request,
    credentials: LoginRequest,
    db: AsyncSession = Depends(get_db),
):
    """
    邮箱密码登录

    - **email**: 用户邮箱
    - **password**: 用户密码
    """
    service = AuthService(db)

    try:
        user = await service.login(credentials)
        tokens = service.create_tokens(user)

        return AuthResponse(
            user=user,
            **tokens,
        )
    except AppException as e:
        raise HTTPException(
            status_code=e.status_code,
            detail={"code": e.code, "message": e.message},
        )


@router.post(
    "/register-email",
    response_model=AuthResponse,
    status_code=status.HTTP_201_CREATED,
    responses={
        409: {"model": ErrorResponse, "description": "邮箱已注册"},
        400: {"model": ErrorResponse, "description": "参数错误"},
    },
)
async def register(
    request: Request,
    data: UserCreate,
    db: AsyncSession = Depends(get_db),
):
    """
    邮箱密码注册

    - **email**: 用户邮箱
    - **password**: 用户密码（至少 8 位，包含字母和数字）
    """

        user = await service.register(data)

            user=user,
            **tokens,

### 8.2 JWT 安全

```python
# backend/app/core/security.py

from datetime import timedelta
from jose import JWTError, jwt

# JWT 配置
SECRET_KEY = "your-secret-key-here"  # 从环境变量读取
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 60 * 24 * 7  # 7 天
REFRESH_TOKEN_EXPIRE_DAYS = 30

def create_access_token(data: dict) -> str:
    """创建访问令牌"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    to_encode.update({"exp": expire, "type": "access"})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

def create_refresh_token(data: dict) -> str:
    """创建刷新令牌"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(days=REFRESH_TOKEN_EXPIRE_DAYS)
    to_encode.update({"exp": expire, "type": "refresh"})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
```
