```python
"""
高德地图 API 集成
"""
import requests
from typing import Optional
from app.config import get_settings

settings = get_settings()


class AmapClient:
    """高德地图 API 客户端"""

    def __init__(self):
        self.api_key = settings.AMAP_API_KEY
        self.base_url = "https://restapi.amap.com/v3"

    def reverse_geocode(
        self,
        lat: float,
        lon: float
    ) -> Optional[dict]:
        """
        逆向地理编码：坐标 → 地址

        Args:
            lat: 纬度
            lon: 经度

        Returns:
            地址信息，失败返回 None
        """
        if not self.api_key:
            print("警告：高德地图 API Key 未配置")
            return None

        url = f"{self.base_url}/geocode/regeo"

        params = {
            "key": self.api_key,
            "location": f"{lon},{lat}",
            "extensions": "base",
            "output": "json"
        }

        try:
            response = requests.get(url, params=params, timeout=5)
            response.raise_for_status()
            data = response.json()

            if data.get("status") == "1":
                return data.get("regeocode")
            else:
                print(f"高德地图 API 错误: {data.get('info')}")
                return None

        except requests.RequestException as e:
            print(f"高德地图 API 请求失败: {e}")
            return None
        except Exception as e:
            print(f"逆向地理编码失败: {e}")
            return None

    def is_configured(self) -> bool:
        """检查是否已配置 API Key"""
        return bool(self.api_key)


# 导出单例
amap_client = AmapClient()
```

```python
"""
简单缓存工具（基于内存）
"""
from typing import TypeVar, Generic, Optional
from datetime import datetime, timedelta
from threading import Lock

T = TypeVar('T')


class CacheEntry(Generic[T]):
    """缓存条目"""
    def __init__(self, value: T, ttl: int):
        self.value = value
        self.expires_at = datetime.utcnow() + timedelta(seconds=ttl)


class SimpleCache:
    """简单内存缓存"""

    def __init__(self):
        self._cache: dict[str, CacheEntry] = {}
        self._lock = Lock()

    def get(self, key: str) -> Optional[T]:
        """获取缓存"""
        with self._lock:
            entry = self._cache.get(key)
            if entry is None:
                return None

            if datetime.utcnow() > entry.expires_at:
                # 过期，删除
                del self._cache[key]
                return None

            return entry.value

    def set(self, key: str, value: T, ttl: int = 3600) -> None:
        """设置缓存"""
        with self._lock:
            self._cache[key] = CacheEntry(value, ttl)

    def delete(self, key: str) -> None:
        """删除缓存"""
        with self._lock:
            self._cache.pop(key, None)

    def clear(self) -> None:
        """清空缓存"""
        with self._lock:
            self._cache.clear()


# 导出单例
```

#### `app/services/geocoding_service.py`

```python
"""
地理编码服务
"""
from typing import Optional
from sqlalchemy.orm import Session

from app.integrations.amap import amap_client
from app.models.event import Event
from app.utils.cache import geocoding_cache
from decimal import Decimal


class GeocodingService:
    """地理编码服务"""

    CACHE_TTL = 86400  # 缓存 24 小时

    def get_location_name(
        self,
        lat: float,
        lon: float
    ) -> Optional[str]:
        """
        获取位置名称（城市+区县）

        Args:
            lat: 纬度
            lon: 经度

        Returns:
            位置名称，如 "浙江省杭州市西湖区"
        """
        # 1. 检查缓存
        cache_key = f"geocode:{lat:.4f},{lon:.4f}"
        cached = geocoding_cache.get(cache_key)
        if cached:
            return cached

        # 2. 调用 API
        if not amap_client.is_configured():
            return None

        result = amap_client.reverse_geocode(lat, lon)
        if not result:
            return None

        # 3. 提取位置信息
        address = result.get("addressComponent", {})
        parts = [
            address.get("province", ""),
            address.get("city", ""),
            address.get("district", "")
        ]
        parts = [p for p in parts if p]

        location_name = "".join(parts)

        # 4. 缓存结果
        geocoding_cache.set(cache_key, location_name, self.CACHE_TTL)

        return location_name

    def update_event_locations(
        self,
        user_id: int,
        db: Session
    ) -> int:
        """
        更新所有事件的位置名称

        Args:
            user_id: 用户 ID
            db: 数据库会话

        Returns:
            更新的数量
        """
        events = db.query(Event).filter(
            Event.user_id == user_id,
            Event.location_name.is_(None),
            Event.gps_lat.isnot(None),
            Event.gps_lon.isnot(None)
        ).all()

        updated = 0

        for event in events:
            lat = float(event.gps_lat)
            lon = float(event.gps_lon)

            location_name = self.get_location_name(lat, lon)
            if location_name:
                event.location_name = location_name
                updated += 1

        db.commit()

        return updated


# 导出服务实例
geocoding_service = GeocodingService()
```
