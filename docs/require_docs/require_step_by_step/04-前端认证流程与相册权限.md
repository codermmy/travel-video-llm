#### 3.3.1 设备工具 `src/utils/deviceUtils.ts`

```typescript
/**
 * 设备相关工具函数
 */
import * as Device from 'expo-device';
import * as Crypto from 'expo-crypto';
import AsyncStorage from '@react-native-async-storage/async-storage';

const DEVICE_ID_KEY = 'travel_album_device_id';

/**
 * 获取或生成设备 ID
 *
 * 逻辑：
 * 1. 先从 AsyncStorage 读取
 * 2. 如果没有，生成新的 UUID 并存储
 * 3. 返回设备 ID
 */
export async function getDeviceId(): Promise<string> {
  try {
    // 1. 尝试从本地读取
    const storedId = await AsyncStorage.getItem(DEVICE_ID_KEY);
    if (storedId) {
      return storedId;
    }

    // 2. 生成新的 UUID
    const randomBytes = Crypto.getRandomBytesAsync(16);
    const bytes = await randomBytes;

    // 转换为 UUID 格式: xxxxxxxx-xxxx-xxxx-xxxxxxxxxxxx
    const uuid = [
      bytes.subarray(0, 4).toHex(),
      bytes.subarray(4, 6).toHex(),
      bytes.subarray(6, 8).toHex(),
      bytes.subarray(8, 10).toHex(),
      bytes.subarray(10, 16).toHex(),
    ].join('-');

    // 3. 存储到本地
    await AsyncStorage.setItem(DEVICE_ID_KEY, uuid);

    return uuid;
  } catch (error) {
    console.error('获取设备 ID 失败:', error);
    // 降级方案：使用设备名称 + 时间戳
    return `device-${Device.deviceName}-${Date.now()}`;
  }
}

/**
 * 获取设备信息（可选，用于统计）
 */
export async function getDeviceInfo(): Promise<{
  deviceName: string;
  deviceType: string;
  platform: string;
}> {
  return {
    deviceName: Device.deviceName || 'Unknown',
    deviceType: Device.deviceType || 'unknown',
    platform: Device.platformApi || 'unknown',
  };
}
```

#### 3.3.3 Token 存储 `src/services/storage/tokenStorage.ts`

```typescript
/**
 * Token 存储管理
 */
import AsyncStorage from '@react-native-async-storage/async-storage';

const TOKEN_KEY = 'travel_album_token';
const USER_ID_KEY = 'travel_album_user_id';
const DEVICE_ID_KEY = 'travel_album_device_id';

export const tokenStorage = {
  /**
   * 保存 Token
   */
  async saveToken(token: string): Promise<void> {
    await AsyncStorage.setItem(TOKEN_KEY, token);
  },

  /**
   * 获取 Token
   */
  async getToken(): Promise<string | null> {
    return await AsyncStorage.getItem(TOKEN_KEY);
  },

  /**
   * 移除 Token
   */
  async removeToken(): Promise<void> {
    await AsyncStorage.removeItem(TOKEN_KEY);
  },

  /**
   * 保存用户 ID
   */
  async saveUserId(userId: number): Promise<void> {
    await AsyncStorage.setItem(USER_ID_KEY, String(userId));
  },

  /**
   * 获取用户 ID
   */
  async getUserId(): Promise<number | null> {
    const userIdStr = await AsyncStorage.getItem(USER_ID_KEY);
    return userIdStr ? parseInt(userIdStr, 10) : null;
  },

  /**
   * 保存设备 ID
   */
  async saveDeviceId(deviceId: string): Promise<void> {
    await AsyncStorage.setItem(DEVICE_ID_KEY, deviceId);
  },

  /**
   * 获取设备 ID
   */
  async getDeviceId(): Promise<string | null> {
    return await AsyncStorage.getItem(DEVICE_ID_KEY);
  },

  /**
   * 清除所有认证数据
   */
  async clearAll(): Promise<void> {
    await AsyncStorage.multiRemove([
      TOKEN_KEY,
      USER_ID_KEY,
      DEVICE_ID_KEY,
    ]);
  },
};
```

#### 3.3.4 完善 API 客户端 `src/services/api/client.ts`

```typescript
/**
 * API 客户端 - 添加 Token 拦截器
 */
import axios, { AxiosError, AxiosInstance } from 'axios';
import { tokenStorage } from '@/services/storage/tokenStorage';
import type { ApiErrorResponse } from '@/types';

// API 基础 URL
const API_BASE_URL = __DEV__
  ? 'http://localhost:8000/api/v1'
  : 'https://api.example.com/api/v1';

/**
 * 创建 Axios 实例
 */
export const apiClient: AxiosInstance = axios.create({
  baseURL: API_BASE_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

/**
 * 请求拦截器：自动添加 Token
 */
apiClient.interceptors.request.use(
  async (config) => {
    const token = await tokenStorage.getToken();
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

/**
 * 响应拦截器：统一错误处理
 */
apiClient.interceptors.response.use(
  (response) => response.data,
  (error: AxiosError<ApiErrorResponse>) => {
    // 网络错误
    if (!error.response) {
      return Promise.reject({
        success: false,
        error: {
          code: 'NETWORK_ERROR',
          message: '网络连接失败，请检查网络设置',
        },
      });
    }

    // HTTP 错误
    const status = error.response.status;
    const errorData = error.response.data;

    // Token 过期，清除本地存储
    if (status === 401) {
      tokenStorage.removeToken();
    }

    return Promise.reject(errorData);
  }
);
```

#### 3.3.5 完善认证 API `src/services/api/authApi.ts`

```typescript
/**
 * 认证相关 API
 */
import { apiClient } from './client';
import { tokenStorage } from '@/services/storage/tokenStorage';
import { getDeviceId } from '@/utils/deviceUtils';
import type { AuthResponse } from '@/types';

interface RegisterRequest {
  deviceId: string;
  nickname?: string;
}

interface RegisterResult {
  success: boolean;
  data?: AuthResponse;
  error?: string;
}

export const authService = {
  /**
   * 设备注册/登录
   */
  async register(nickname?: string): Promise<RegisterResult> {
    try {
      // 获取设备 ID
      const deviceId = await getDeviceId();

      // 调用注册接口
      const response = await apiClient<{
        success: true;
        data: AuthResponse;
        timestamp: string;
      }>({
        method: 'POST',
        url: '/auth/register',
        data: { deviceId, nickname },
      });

      // 保存 Token 和用户信息
      await tokenStorage.saveToken(response.data.token);
      await tokenStorage.saveUserId(response.data.userId);
      await tokenStorage.saveDeviceId(response.data.deviceId);

      return {
        success: true,
        data: response.data,
      };
    } catch (error: any) {
      return {
        success: false,
        error: error.error?.message || '注册失败，请重试',
      };
    }
  },

  /**
   * 检查是否已登录
   */
  async isAuthenticated(): Promise<boolean> {
    const token = await tokenStorage.getToken();
    return !!token;
  },

  /**
   * 登出
   */
  async logout(): Promise<void> {
    await tokenStorage.clearAll();
  },

  /**
   * 获取本地用户信息
   */
  async getLocalUserInfo(): Promise<{
    userId: number | null;
    deviceId: string | null;
  }> {
    const [userId, deviceId] = await Promise.all([
      tokenStorage.getUserId(),
      tokenStorage.getDeviceId(),
    ]);

    return { userId, deviceId };
  },
};
```

#### 3.3.6 完善认证状态 `src/stores/authStore.ts`

```typescript
/**
 * 认证状态管理
 */
import { create } from 'zustand';
import { authService } from '@/services/api/authApi';
import type { User } from '@/types';

interface AuthState {
  // 状态
  user: User | null;
  token: string | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  isNewUser: boolean;
  error: string | null;

  // 操作
  register: (nickname?: string) => Promise<boolean>;
  logout: () => Promise<void>;
  checkAuth: () => Promise<void>;
  clearError: () => void;
}

export const useAuthStore = create<AuthState>((set) => ({
  // 初始状态
  user: null,
  token: null,
  isAuthenticated: false,
  isLoading: false,
  isNewUser: false,
  error: null,

  // 注册/登录
  register: async (nickname?: string) => {
    set({ isLoading: true, error: null });

    try {
      const result = await authService.register(nickname);

      if (result.success && result.data) {
        set({
          user: {
            userId: result.data.userId,
            deviceId: result.data.deviceId,
            nickname: result.data.nickname,
            createdAt: result.data.createdAt,
          },
          token: result.data.token,
          isAuthenticated: true,
          isNewUser: result.data.isNewUser,
          isLoading: false,
        });
        return true;
      } else {
        set({
          error: result.error || '注册失败',
          isLoading: false,
        });
        return false;
      }
    } catch (error: any) {
      set({
        error: error.message || '网络错误',
        isLoading: false,
      });
      return false;
    }
  },

  // 登出
  logout: async () => {
    await authService.logout();
    set({
      user: null,
      token: null,
      isAuthenticated: false,
      isNewUser: false,
    });
  },

  // 检查认证状态
  checkAuth: async () => {
    set({ isLoading: true });

    const isAuthenticated = await authService.isAuthenticated();

    if (isAuthenticated) {
      // 从本地恢复用户信息
      const { userId, deviceId } = await authService.getLocalUserInfo();

      set({
        user: userId
          ? {
              userId,
              deviceId: deviceId || '',
              nickname: '旅行者',
              createdAt: new Date().toISOString(),
            }
          : null,
        isAuthenticated: true,
        isLoading: false,
      });
    } else {
      set({
        user: null,
        isAuthenticated: false,
        isLoading: false,
      });
    }
  },

  // 清除错误
  clearError: () => set({ error: null }),
}));
```
